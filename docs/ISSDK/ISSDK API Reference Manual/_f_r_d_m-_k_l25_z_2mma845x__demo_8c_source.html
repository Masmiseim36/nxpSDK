116 /* Handler for Device Info and Streaming Control Commands. */
117 bool process_host_command(
118  uint8_t tag, uint8_t *hostCommand, uint8_t *hostResponse, size_t *hostMsgSize, size_t respBufferSize)
119 {
120  bool success = false;
121 
122  /* If it is Host requesting Device Info, send Board Name and Shield Name. */
123  if (tag == HOST_PRO_INT_DEV_TAG)
124  { /* Byte 1 : Payload - Length of APPLICATION_NAME (b)
125  * Bytes=b : Payload Application Name
126  * Byte b+1 : Payload - Length of BOARD_NAME (s)
127  * Bytes=s : Payload Board Name
128  * Byte b+s+2 : Payload - Length of SHIELD_NAME (v)
129  * Bytes=v : Payload Shield Name */
130 
131  size_t appNameLen = strlen(embAppName);
132  size_t boardNameLen = strlen(boardString);
133  size_t shieldNameLen = strlen(shieldString);
134 
135  if (respBufferSize >= boardNameLen + shieldNameLen + appNameLen + 3)
136  { /* We have sufficient buffer. */
137  *hostMsgSize = 0;
138  }
139  else
140  {
141  return false;
142  }
143 
144  hostResponse[*hostMsgSize] = appNameLen;
145  *hostMsgSize += 1;
146 
147  memcpy(hostResponse + *hostMsgSize, embAppName, appNameLen);
148  *hostMsgSize += appNameLen;
149 
150  hostResponse[*hostMsgSize] = boardNameLen;
151  *hostMsgSize += 1;
152 
153  memcpy(hostResponse + *hostMsgSize, boardString, boardNameLen);
154  *hostMsgSize += boardNameLen;
155 
156  hostResponse[*hostMsgSize] = shieldNameLen;
157  *hostMsgSize += 1;
158 
159  memcpy(hostResponse + *hostMsgSize, shieldString, shieldNameLen);
160  *hostMsgSize += shieldNameLen;
161 
162  return true;
163  }
164 
165  /* If it is Host sending Streaming Commands, take necessary actions. */
166  if ((tag == (HOST_PRO_INT_CMD_TAG | HOST_PRO_CMD_W_CFG_TAG)) &&
167  (*hostMsgSize == HOST_MSG_CMD_ACT_OFFSET - HOST_MSG_LEN_LSB_OFFSET))
168  { /* Byte 1 : Payload - Operation Code (Start/Stop Operation Code)
169  * Byte 2 : Payload - Stream ID (Target Stream for carrying out operation) */
170  switch (hostCommand[0]) /* Execute desired operation (Start/Stop) on the requested Stream. */
171  {
172  case HOST_CMD_START:
173  if (hostCommand[1] == gStreamID && bMma845xReady && bStreamingEnabled == false)
174  {
175  bStreamingEnabled = true;
176  success = true;
177  }
178  break;
179  case HOST_CMD_STOP:
180  if (hostCommand[1] == gStreamID && bMma845xReady && bStreamingEnabled == true)
181  {
182  pGpioDriver->clr_pin(&GREEN_LED);
183  bStreamingEnabled = false;
184  success = true;
185  }
186  break;
187  default:
188  break;
189  }
190  *hostMsgSize = 0; /* Zero payload in response. */
191  }
192 
193  return success;
194 }

196 /*!
197  * @brief Main function
198  */
199 int main(void)
200 {
201  int32_t status, systick;
202  uint8_t data[MMA845x_ACCEL_DATA_SIZE], streamingPacket[STREAMING_HEADER_LEN + MMA845x_STREAM_DATA_SIZE];
203 
204  mma845x_i2c_sensorhandle_t mma845xDriver;
205  mma845x_acceldata_t rawData = {.timestamp = 0};
206 
207  ARM_DRIVER_I2C *pI2cDriver = &I2C_S_DRIVER;
208  ARM_DRIVER_USART *pUartDriver = &HOST_S_DRIVER;
209 
210  /*! Initialize the MCU hardware. */
211  BOARD_BootClockRUN();
212  BOARD_SystickEnable();
213  CLOCK_SetLpsci0Clock(1); /* Start the LPSCI Clock (01 : IRC48M). */
214 
215  /* Create the Short Application Name String for ADS. */
216  sprintf(embAppName, "%s:%s", APPLICATION_NAME, APPLICATION_VERSION);
217 
218  /* Run ADS. */
219  BOARD_RunADS(embAppName, boardString, shieldString, ADS_MAX_STRING_LENGTH);
220 
221  /* Create the Full Application Name String for Device Info Response. */
222  sprintf(embAppName, "%s:%s:%s", SHIELD_NAME, APPLICATION_NAME, APPLICATION_VERSION);
223 
224  /*! Initialize INT1 MMA845x pin used by FRDM board */
225  pGpioDriver->pin_init(&MMA845x_INT1, GPIO_DIRECTION_IN, NULL, &mma845x_int_data_ready_callback, NULL);
226 
227  /*! Initialize GREEN LED pin used by FRDM board */
228  pGpioDriver->pin_init(&GREEN_LED, GPIO_DIRECTION_OUT, NULL, NULL, NULL);
229 
230  /*! Initialize the I2C driver. */
231  status = pI2cDriver->Initialize(I2C_S_SIGNAL_EVENT);
232  if (ARM_DRIVER_OK != status)
233  {
234  return -1;
235  }
236 
237  /*! Set the I2C Power mode. */
238  status = pI2cDriver->PowerControl(ARM_POWER_FULL);
239  if (ARM_DRIVER_OK != status)
240  {
241  return -1;
242  }
243 
244  /*! Set the I2C bus speed. */
245  status = pI2cDriver->Control(ARM_I2C_BUS_SPEED, ARM_I2C_BUS_SPEED_FAST);
246  if (ARM_DRIVER_OK != status)
247  {
248  return -1;
249  }
250 
251  /*! Initialize the UART driver. */
252  status = pUartDriver->Initialize(HOST_S_SIGNAL_EVENT);
253  if (ARM_DRIVER_OK != status)
254  {
255  return -1;
256  }
257 
258  /*! Set the UART Power mode. */
259  status = pUartDriver->PowerControl(ARM_POWER_FULL);
260  if (ARM_DRIVER_OK != status)
261  {
262  return -1;
263  }
264 
265  /*! Set UART Baud Rate. */
266  status = pUartDriver->Control(ARM_USART_MODE_ASYNCHRONOUS, BOARD_DEBUG_UART_BAUDRATE);
267  if (ARM_DRIVER_OK != status)
268  {
269  return -1;
270  }
271 
272  /*! Initialize the MMA845x sensor driver. */
273  status = MMA845x_I2C_Initialize(&mma845xDriver, &I2C_S_DRIVER, I2C_S_DEVICE_INDEX, MMA845x_I2C_ADDR,
274  MMA8451_WHO_AM_I_WHOAMI_VALUE);
275  if (SENSOR_ERROR_NONE == status)
276  {
277  /*! Set the task to be executed while waiting for I2C transactions to complete. */
278  MMA845x_I2C_SetIdleTask(&mma845xDriver, (registeridlefunction_t)SMC_SetPowerModeWait, SMC);
279 
280  /*! Configure the MMA845x sensor driver. */
281  status = MMA845x_I2C_Configure(&mma845xDriver, mma845x_Config_Isr);
282  if (SENSOR_ERROR_NONE == status)
283  {
284  bMma845xReady = true;
285  }
286  }
287 
288  /*! Initialize streaming and assign a Stream ID. */
289  gStreamID =
290  Host_IO_Init(pUartDriver, (void *)mma845xDriver.pCommDrv, &mma845xDriver.deviceInfo, NULL, MMA845x_I2C_ADDR);
291  /* Confirm if a valid Stream ID has been allocated for this stream. */
292  if (0 == gStreamID)
293  {
294  bMma845xReady = false;
295  }
296  else
297  {
298  /*! Populate streaming header. */
299  Host_IO_Add_ISO_Header(gStreamID, streamingPacket, MMA845x_STREAM_DATA_SIZE);
300  BOARD_SystickStart(&systick);
301  pGpioDriver->clr_pin(&GREEN_LED); /* Set LED to indicate application is ready. */
302  }
303 
304  for (;;) /* Forever loop */
305  { /* Call UART Non-Blocking Receive. */
306  Host_IO_Receive(process_host_command, HOST_FORMAT_HDLC);
307 
308  /* Process packets only if streaming has been enabled by Host and ISR is available.
309  * In ISR Mode we do not need to check Data Ready Register.
310  * The receipt of interrupt will indicate data is ready. */
311  if (false == bStreamingEnabled || false == bMma845xDataReady)
312  {
313  SMC_SetPowerModeWait(SMC); /* Power save, wait if nothing to do. */
314  continue;
315  }
316  else
317  { /*! Clear the data ready flag, it will be set again by the ISR. */
318  bMma845xDataReady = false;
319  pGpioDriver->toggle_pin(&GREEN_LED);
320  }
321 
322  /*! Read raw sensor data from the MMA845x. */
323  status = MMA845x_I2C_ReadData(&mma845xDriver, mma845x_Output_Values, data);
324  if (ARM_DRIVER_OK != status)
325  { /* Loop, if sample read failed. */
326  continue;
327  }
328 
329  /* Update timestamp from Systick framework. */
330  rawData.timestamp += BOARD_SystickElapsedTime_us(&systick);
331 
332  /*! Convert the raw sensor data to signed 32-bit and 16-bit containers for display to the debug port. */
333  rawData.accel[0] = ((int16_t)data[0] << 8) | data[1];
334  rawData.accel[0] /= 4;
335  rawData.accel[1] = ((int16_t)data[2] << 8) | data[3];
336  rawData.accel[1] /= 4;
337  rawData.accel[2] = ((int16_t)data[4] << 8) | data[5];
338  rawData.accel[2] /= 4;
339 
340  /* Copy Raw samples to Streaming Buffer. */
341  memcpy(streamingPacket + STREAMING_HEADER_LEN, &rawData, MMA845x_STREAM_DATA_SIZE);
342  /* Send streaming packet to Host. */
343  Host_IO_Send(streamingPacket, sizeof(streamingPacket), HOST_FORMAT_HDLC);
344  }
345 }
