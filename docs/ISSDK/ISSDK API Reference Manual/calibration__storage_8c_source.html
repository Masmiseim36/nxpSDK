43 void SaveMagCalibrationToNVM(SensorFusionGlobals *sfg)
44 {
45 #if F_USING_MAG
46  uint8_t *pSrc, *pDst; // scratch pointers
47  int16_t i; // loop counter
48  uint8_t iNVMBuffer[256]; // NVM write buffer (smallest size writeable to flash)
49  uint32_t itmp32;
50 
51  // copy existing magnetic, gyro and accelerometer calibrations to buffer
52  pSrc = (uint8 *) CALIBRATION_NVM_ADDR;
53  pDst = iNVMBuffer;
54  for (i = 0; i < 256; i++)
55  *(pDst++) = *(pSrc++);
56  // default to no magnetic calibration in header
57  iNVMBuffer[MAG_NVM_OFFSET] = iNVMBuffer[MAG_NVM_OFFSET + 1] = iNVMBuffer[MAG_NVM_OFFSET + 2] = iNVMBuffer[MAG_NVM_OFFSET + 3] = 0xFF;
58 
59  // fill the buffer with the magnetic calibration in bytes 0 to 67 (total 68 bytes)
60  // [0-3]: four byte header denoting magnetic calibration present
61  itmp32 = 0x12345678;
62  pSrc = (uint8 *) &itmp32;
63  pDst = iNVMBuffer + MAG_NVM_OFFSET;
64  for (i = 0; i < 4; i++)
65  *(pDst++) = *(pSrc++);
66  // [4-67]: magnetic calibration: 15x float + 1x int32 total 64 bytes
67  pSrc = (uint8 *) &(sfg->MagCal);
68  for (i = 0; i < 64; i++)
69  *(pDst++) = *(pSrc++);
70 
71  // write the whole buffer contents to NVM
72  NVM_SetBlockFlash(iNVMBuffer, CALIBRATION_NVM_ADDR, 256);
73 #endif // if F_USING_MAG
74  return;
75 }
