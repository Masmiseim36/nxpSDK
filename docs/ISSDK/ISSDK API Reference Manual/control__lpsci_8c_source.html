31 /*! \file control_lpsci.c
32  \brief Defines control sub-system
33 
34  This file contains a Low power UART implementation of the control subsystem.
35  This version is targeted specificially at FRDM-KL25Z, which utilizes a low
36  power uart to drive both the OpenSDA and shield UART connections for
37  FRDM-MULT2-B Bluetooth module.
38 
39  The low power uart utilizes a slightly different interface within KSDK, hence
40  this adaptation.
41 
42  The command interpreter and streaming functions are contained in two separate
43  files. So you can easily swap those out with only minor changes here.
44 */

65 // Blocking function to write a single byte to a specified UART
66 void myUART_WriteByte(UART0_Type *base, uint8_t data)
67 {
68  uint8_t flag = 0;
69  while (!flag)
70  {
71  flag = base->S1 & 0x80;
72  }
73  LPSCI_WriteByte(base, data);
74 }

76 // Blocking function pipes specified buffer to both output UARTS
77 int8_t writeControlPort(ControlSubsystem *pComm, uint8_t buffer[], uint16_t nbytes)
78 {
79  uint16_t i;
80  for (i = 0; i < nbytes; i++)
81  {
82  myUART_WriteByte(CONTROL_UART, buffer[i]);
83  }
84 
85  return (0);
86 }

88 // initialize BlueRadios BR-LE4.0-D2A Bluetooth module
89 // This is required for NXP FRDM-FXS-MULT2-B boards.
90 void BlueRadios_Init(void)
91 {
92  uint16_t ilen; // command string length
93 
94  // transmit "ATSRM,2,0\r" to minimize traffic from the module
95  // command "ATSRM": sets the module response mode which configures how verbose the module will be
96  // 2: response mode at to minimal
97  // 0: disconnected mode is command mode
98  // \r: carriage return escape sequence
99  strcpy((char *)sUARTOutputBuffer, "ATSRM,2,0\r");
100  ilen = strlen((char *)sUARTOutputBuffer);
101  writeControlPort(NULL, sUARTOutputBuffer, ilen);
102  return;
103 }

105 void CONTROL_UART_IRQHandler(void)
106 {
107  uint8_t data;
108  static char iCommandBuffer_A[5] = "~~~~"; // 5 bytes long to include the unused terminating \0
109 
110  sfg.setStatus(&sfg, RECEIVING_WIRED);
111  if ((kLPSCI_RxDataRegFullFlag)&LPSCI_GetStatusFlags(CONTROL_UART))
112  {
113  data = LPSCI_ReadByte(CONTROL_UART);
114  DecodeCommandBytes(&sfg, iCommandBuffer_A, &data, 1);
115  }
116 }

118 /// Initialize the control subsystem and all related hardware
119 int8_t initializeControlPort(
120  ControlSubsystem *pComm ///< pointer to the control subystem structure
121 )
122 {
123  lpsci_config_t config;
124  if (pComm)
125  {
126  pComm->DefaultQuaternionPacketType = Q3; // default to simplest algorithm
127  pComm->QuaternionPacketType = Q3; // default to simplest algorithm
128  pComm->AngularVelocityPacketOn = true; // transmit angular velocity packet
129  pComm->DebugPacketOn = true; // transmit debug packet
130  pComm->RPCPacketOn = true; // transmit roll, pitch, compass packet
131  pComm->AltPacketOn = true; // Altitude packet
132  pComm->AccelCalPacketOn = 0;
133  pComm->write = writeControlPort;
134  pComm->stream = CreateAndSendPackets;
135 
136  /* Initialize WIRED UART pins below - currently duplicates code in pin_mux.c */
137  CLOCK_SetLpsci0Clock(0x1U);
138  CLOCK_EnableClock(CONTROL_UART_PORT_CLKEN);
139  PORT_SetPinMux(CONTROL_UART_PORT, CONTROL_UART_RX_PIN, CONTROL_UART_MUX);
140  PORT_SetPinMux(CONTROL_UART_PORT, CONTROL_UART_TX_PIN, CONTROL_UART_MUX);
141  LPSCI_GetDefaultConfig(&config);
142 
143  config.baudRate_Bps = CONTROL_BAUDRATE;
144  config.enableTx = true;
145  config.enableRx = true;
146  LPSCI_Init(CONTROL_UART, &config, CLOCK_GetFreq(CONTROL_UART_CLKSRC));
147  BlueRadios_Init();
148 
149  /* Enable RX interrupt. */
150  LPSCI_EnableInterrupts(CONTROL_UART, kLPSCI_RxDataRegFullInterruptEnable);
151  EnableIRQ(CONTROL_UART_IRQn);
152  return (0);
153  }
154  else
155  {
156  return (1);
157  }
158 }
