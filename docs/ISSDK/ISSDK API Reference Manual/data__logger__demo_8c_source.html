168 /* Handler for Device Info and Streaming Control Commands. */
169 bool process_host_command(
170  uint8_t tag, uint8_t *hostCommand, uint8_t *hostResponse, size_t *hostMsgSize, size_t respBufferSize)
171 {
172  bool success = false;
173 
174  /* If it is Host requesting Device Info, send Board Name and Shield Name. */
175  if (tag == HOST_PRO_INT_DEV_TAG)
176  { /* Byte 1 : Payload - Length of APPLICATION_NAME (b)
177  * Bytes=b : Payload Application Name
178  * Byte b+1 : Payload - Length of BOARD_NAME (s)
179  * Bytes=s : Payload Board Name
180  * Byte b+s+2 : Payload - Length of SHIELD_NAME (v)
181  * Bytes=v : Payload Shield Name */
182 
183  size_t appNameLen = strlen(embAppName);
184  size_t boardNameLen = strlen(boardString);
185  size_t shieldNameLen = strlen(shieldString);
186 
187  if (respBufferSize >= boardNameLen + shieldNameLen + appNameLen + 3)
188  { /* We have sufficient buffer. */
189  *hostMsgSize = 0;
190  }
191  else
192  {
193  return false;
194  }
195 
196  hostResponse[*hostMsgSize] = appNameLen;
197  *hostMsgSize += 1;
198 
199  memcpy(hostResponse + *hostMsgSize, embAppName, appNameLen);
200  *hostMsgSize += appNameLen;
201 
202  hostResponse[*hostMsgSize] = boardNameLen;
203  *hostMsgSize += 1;
204 
205  memcpy(hostResponse + *hostMsgSize, boardString, boardNameLen);
206  *hostMsgSize += boardNameLen;
207 
208  hostResponse[*hostMsgSize] = shieldNameLen;
209  *hostMsgSize += 1;
210 
211  memcpy(hostResponse + *hostMsgSize, shieldString, shieldNameLen);
212  *hostMsgSize += shieldNameLen;
213 
214  return true;
215  }
216 
217  /* If it is Host sending Streaming Commands, take necessary actions. */
218  if ((tag == (HOST_PRO_INT_CMD_TAG | HOST_PRO_CMD_W_CFG_TAG)) &&
219  (*hostMsgSize == HOST_MSG_CMD_ACT_OFFSET - HOST_MSG_LEN_LSB_OFFSET))
220  { /* Byte 1 : Payload - Operation Code (Start/Stop Operation Code)
221  * Byte 2 : Payload - Stream ID (Target Stream for carrying out operation) */
222  switch (hostCommand[0]) /* Execute desired operation (Start/Stop) on the requested Stream. */
223  {
224  case HOST_CMD_START:
225  if (hostCommand[1] == gPrimaryStreamID && bDataLoggerReady && bStreamingEnabled == false)
226  {
227  bStreamingEnabled = true;
228  success = true;
229  }
230  break;
231  case HOST_CMD_STOP:
232  if (hostCommand[1] == gPrimaryStreamID && bDataLoggerReady && bStreamingEnabled == true)
233  {
234  pGpioDriver->clr_pin(&LED_GREEN);
235  bStreamingEnabled = false;
236  success = true;
237  }
238  break;
239  default:
240  break;
241  }
242  *hostMsgSize = 0; /* Zero payload in response. */
243  }
244 
245  return success;
246 }

251 int main(void)
252 {
253  size_t payLoadLen;
254  int32_t status, systick;
255  uint8_t secondaryStreamID1, secondaryStreamID2, /* The auto assigned Stream ID not to be used to stream data. */
256  dataReady_3115, toggle_pin = 0, /* The MPL3115 sensor data ready flag and LED Toggle Counter. */
257  data[FXOS8700_DATA_SIZE + FXAS21002_DATA_SIZE + MPL3115_DATA_SIZE],
258  streamingPacket[STREAMING_HEADER_LEN + STREAMING_PAYLOAD_LEN];
259 
260  fxos8700_accelmagdata_t rawData_fxos8700;
261  fxas21002_gyrodata_t rawData_fxas21002;
262  mpl3115_pressuredata_t rawData_mpl3115;
263 
264  ARM_DRIVER_I2C *pI2cDriver = &I2C_S_DRIVER;
265  ARM_DRIVER_USART *pUartDriver = &HOST_S_DRIVER;
266 
267  mpl3115_i2c_sensorhandle_t mpl3115Driver;
268  fxos8700_i2c_sensorhandle_t fxos8700Driver;
269  fxas21002_i2c_sensorhandle_t fxas21002Driver;
270 
271  /*! Initialize the MCU hardware. */
272  BOARD_BootClockRUN();
273  BOARD_SystickEnable();
274  CLOCK_SetLpsci0Clock(1); /* Start the LPSCI Clock (01 : IRC48M). */
275 
276  /* Create the Short Application Name String for ADS. */
277  sprintf(embAppName, "%s:%s", APPLICATION_NAME, APPLICATION_VERSION);
278 
279  /* Run ADS. */
280  BOARD_RunADS(embAppName, boardString, shieldString, ADS_MAX_STRING_LENGTH);
281 
282  /* Create the Full Application Name String for Device Info Response. */
283  sprintf(embAppName, "%s:%s:%s", SHIELD_NAME, APPLICATION_NAME, APPLICATION_VERSION);
284 
285  /*! Initialize INT1 FXAS21002 pin used by RD board */
286  pGpioDriver->pin_init(&INT1_FXAS21002, GPIO_DIRECTION_IN, NULL, &fxas21002_isr, NULL);
287 
288  /*! Initialize GREEN LED pin used by RD board */
289  pGpioDriver->pin_init(&LED_GREEN, GPIO_DIRECTION_OUT, NULL, NULL, NULL);
290  pGpioDriver->set_pin(&LED_GREEN); /* Clear LED to indicate application is not ready. */
291 
292  /*! Initialize the I2C driver. */
293  status = pI2cDriver->Initialize(I2C_S_SIGNAL_EVENT);
294  if (ARM_DRIVER_OK != status)
295  {
296  return -1;
297  }
298 
299  /*! Set the I2C Power mode. */
300  status = pI2cDriver->PowerControl(ARM_POWER_FULL);
301  if (ARM_DRIVER_OK != status)
302  {
303  return -1;
304  }
305 
306  /*! Set the I2C bus speed. */
307  status = pI2cDriver->Control(ARM_I2C_BUS_SPEED, ARM_I2C_BUS_SPEED_FAST);
308  if (ARM_DRIVER_OK != status)
309  {
310  return -1;
311  }
312 
313  /*! Initialize the UART driver. */
314  status = pUartDriver->Initialize(HOST_S_SIGNAL_EVENT);
315  if (ARM_DRIVER_OK != status)
316  {
317  return -1;
318  }
319 
320  /*! Set the UART Power mode. */
321  status = pUartDriver->PowerControl(ARM_POWER_FULL);
322  if (ARM_DRIVER_OK != status)
323  {
324  return -1;
325  }
326 
327  /*! Set UART Baud Rate. */
328  status = pUartDriver->Control(ARM_USART_MODE_ASYNCHRONOUS, BOARD_DEBUG_UART_BAUDRATE);
329  if (ARM_DRIVER_OK != status)
330  {
331  return -1;
332  }
333 
334  do
335  {
336  /*! Initialize the MPL3115 sensor driver. */
337  status = MPL3115_I2C_Initialize(&mpl3115Driver, &I2C_S_DRIVER, I2C_S_DEVICE_INDEX, MPL3115_I2C_ADDR,
338  MPL3115_WHOAMI_VALUE);
339  if (SENSOR_ERROR_NONE != status)
340  {
341  break;
342  }
343  /*! Initialize the FXOS8700 sensor driver. */
344  status = FXOS8700_I2C_Initialize(&fxos8700Driver, &I2C_S_DRIVER, I2C_S_DEVICE_INDEX, FXOS8700_I2C_ADDR,
345  FXOS8700_WHO_AM_I_PROD_VALUE);
346  if (SENSOR_ERROR_NONE != status)
347  {
348  break;
349  }
350  /*! Initialize the FXAS21002 sensor driver. */
351  status = FXAS21002_I2C_Initialize(&fxas21002Driver, &I2C_S_DRIVER, I2C_S_DEVICE_INDEX, FXAS21002_I2C_ADDR,
352  FXAS21002_WHO_AM_I_WHOAMI_PROD_VALUE);
353  if (SENSOR_ERROR_NONE != status)
354  {
355  break;
356  }
357 
358  /*! Set the task to be executed while waiting for I2C transactions to complete. */
359  MPL3115_I2C_SetIdleTask(&mpl3115Driver, (registeridlefunction_t)SMC_SetPowerModeWait, SMC);
360  FXOS8700_I2C_SetIdleTask(&fxos8700Driver, (registeridlefunction_t)SMC_SetPowerModeWait, SMC);
361  FXAS21002_I2C_SetIdleTask(&fxas21002Driver, (registeridlefunction_t)SMC_SetPowerModeWait, SMC);
362 
363  /*! Configure the fxos8700 sensor driver. */
364  status = FXOS8700_I2C_Configure(&fxos8700Driver, fxos8700_Config_Hybrid);
365  if (SENSOR_ERROR_NONE != status)
366  {
367  break;
368  }
369  /*! Configure the FXAS21002 sensor driver. */
370  status = FXAS21002_I2C_Configure(&fxas21002Driver, fxas21002_Config_Isr);
371  if (SENSOR_ERROR_NONE != status)
372  {
373  break;
374  }
375  /*! In One-Shot Mode we do not need to Configure MPL3115, instead we will set OST bit directly. */
376 
377  /* If we reach here then all sensors have been initialized, configured and GDL is ready. */
378  bDataLoggerReady = true;
379 
380  } while (false);
381 
382  /*! Initialize streaming and assign a Stream IDs. */
383  gPrimaryStreamID =
384  Host_IO_Init(pUartDriver, (void *)mpl3115Driver.pCommDrv, &mpl3115Driver.deviceInfo, NULL, MPL3115_I2C_ADDR);
385  /* Confirm if a valid Stream ID has been allocated for this stream. */
386  if (0 == gPrimaryStreamID)
387  {
388  bDataLoggerReady = false;
389  }
390  secondaryStreamID1 = /* This is required for registering the slave address with Host I/O for Register Read/Write. */
391  Host_IO_Init(pUartDriver, (void *)fxos8700Driver.pCommDrv, &fxos8700Driver.deviceInfo, NULL, FXOS8700_I2C_ADDR);
392  /* Confirm if a valid Stream ID has been allocated for this stream. */
393  if (0 == secondaryStreamID1)
394  {
395  bDataLoggerReady = false;
396  }
397  secondaryStreamID2 = /* This is required for registering the slave address with Host I/O for Register Read/Write. */
398  Host_IO_Init(pUartDriver, (void *)fxas21002Driver.pCommDrv, &fxas21002Driver.deviceInfo, NULL,
399  FXAS21002_I2C_ADDR);
400  /* Confirm if a valid Stream ID has been allocated for this stream. */
401  if (0 == secondaryStreamID2)
402  {
403  bDataLoggerReady = false;
404  }
405 
406  if (true == bDataLoggerReady)
407  {
408  *((uint32_t *)&streamingPacket[STREAMING_HEADER_LEN]) = 0; /* Initialize time stamp field. */
409  BOARD_SystickStart(&systick);
410  pGpioDriver->clr_pin(&LED_GREEN); /* Set LED to indicate application is ready. */
411  }
412 
413  for (;;) /* Forever loop */
414  { /* Check for incoming commands from Host. */
415  Host_IO_Receive(process_host_command, HOST_FORMAT_HDLC);
416 
417  /* Process packets only if streaming has been enabled by Host and ISR is available.
418  * In ISR Mode we do not need to check Data Ready Register.
419  * The receipt of interrupt will indicate data is ready. */
420  if (false == bStreamingEnabled || false == bFxas21002DataReady)
421  {
422  SMC_SetPowerModeWait(SMC); /* Power save, wait if nothing to do. */
423  continue;
424  }
425  else
426  { /*! Clear the data ready flag, it will be set again by the ISR. */
427  bFxas21002DataReady = false;
428  }
429 
430  /* Read timestamp from Systick framework. */
431  *((uint32_t *)&streamingPacket[STREAMING_HEADER_LEN]) += BOARD_SystickElapsedTime_us(&systick);
432 
433  /*! Read the raw sensor data from the fxos8700. */
434  status = FXOS8700_I2C_ReadData(&fxos8700Driver, fxos8700_Output_values, data);
435  if (ARM_DRIVER_OK != status)
436  {
437  return -1;
438  }
439 
440  /* Convert to Little Endian, Right Justified, Even Padded Signed integer counts. */
441  rawData_fxos8700.accel[0] = ((int16_t)data[0] << 8) | data[1];
442  rawData_fxos8700.accel[0] /= 4;
443  rawData_fxos8700.accel[1] = ((int16_t)data[2] << 8) | data[3];
444  rawData_fxos8700.accel[1] /= 4;
445  rawData_fxos8700.accel[2] = ((int16_t)data[4] << 8) | data[5];
446  rawData_fxos8700.accel[2] /= 4;
447  rawData_fxos8700.mag[0] = ((int16_t)data[6] << 8) | data[7];
448  rawData_fxos8700.mag[1] = ((int16_t)data[8] << 8) | data[9];
449  rawData_fxos8700.mag[2] = ((int16_t)data[10] << 8) | data[11];
450 
451  /* Copy the converted sample to the streaming buffer. */
452  memcpy(streamingPacket + STREAMING_HEADER_LEN + STREAMING_PKT_TIMESTAMP_LEN, &rawData_fxos8700.accel,
453  sizeof(rawData_fxos8700.accel));
454  memcpy(streamingPacket + STREAMING_HEADER_LEN + STREAMING_PKT_TIMESTAMP_LEN + FXOS8700_ACCEL_DATA_SIZE,
455  &rawData_fxos8700.mag, sizeof(rawData_fxos8700.mag));
456 
457  /*! Read the raw sensor data from the FXAS21002. */
458  status = FXAS21002_I2C_ReadData(&fxas21002Driver, fxas21002_Output_Values, data + FXOS8700_DATA_SIZE);
459  if (ARM_DRIVER_OK != status)
460  {
461  return -1;
462  }
463 
464  /* Convert to Little Endian, Right Justified, Even Padded Signed integer counts. */
465  rawData_fxas21002.gyro[0] = ((int16_t)data[12] << 8) | data[13];
466  rawData_fxas21002.gyro[1] = ((int16_t)data[14] << 8) | data[15];
467  rawData_fxas21002.gyro[2] = ((int16_t)data[16] << 8) | data[17];
468 
469  /* Copy the converted sample to the streaming buffer. */
470  memcpy(streamingPacket + STREAMING_HEADER_LEN + STREAMING_PKT_TIMESTAMP_LEN + FXOS8700_DATA_SIZE,
471  &rawData_fxas21002.gyro, sizeof(rawData_fxas21002.gyro));
472 
473  /*! MPL3115 is in One-Shot Mode so we have to trigger acquizition of new sample based on OST bit. */
474  status = MPL3115_I2C_ReadData(&mpl3115Driver, cMpl3115GetOST, &dataReady_3115);
475  if (ARM_DRIVER_OK != status)
476  {
477  return -1;
478  }
479  if (MPL3115_CTRL_REG1_OST_RESET == (dataReady_3115 & MPL3115_CTRL_REG1_OST_MASK))
480  {
481  /*! Read raw sensor data from the MPL3115. */
482  status = MPL3115_I2C_ReadData(&mpl3115Driver, mpl3115_Output_Values,
483  data + FXOS8700_DATA_SIZE + FXAS21002_DATA_SIZE);
484  if (ARM_DRIVER_OK != status)
485  {
486  return -1;
487  }
488 
489  /* Convert to Little Endian, Right Justified, Even Padded Signed integer counts. */
490  rawData_mpl3115.pressure = (uint32_t)((data[18]) << 16) | ((data[19]) << 8) | ((data[20]));
491  rawData_mpl3115.pressure /= 16;
492  rawData_mpl3115.temperature = (int16_t)((data[21]) << 8) | (data[22]);
493  rawData_mpl3115.temperature /= 16;
494 
495  /* Copy the converted sample to the streaming buffer. */
496  memcpy(streamingPacket + STREAMING_HEADER_LEN + STREAMING_PKT_TIMESTAMP_LEN + FXOS8700_DATA_SIZE +
497  FXAS21002_DATA_SIZE,
498  &rawData_mpl3115.pressure, sizeof(rawData_mpl3115.pressure));
