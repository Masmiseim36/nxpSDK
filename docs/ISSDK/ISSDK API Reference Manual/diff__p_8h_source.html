119  /* data acquisition is completed.. */
120

122  /* Temperature acquisition is completed before the TDR flag is cleared. */
123 
124  uint8_t pow : 1; /* Pressure Data Overwrite. POW is set to logic 1 whenever a new Pressure */
125  /* acquisition is completed before the PDR flag has been cleared. */
