91 void KE15Z_LPTMR0_ADC0_Initialize(void)
92 {
93  lptmr_config_t lptmrConfig;
94  adc12_config_t adc12ConfigStruct;
95 
96  /* Configure ODR Timer. */
97  LPTMR_GetDefaultConfig(&lptmrConfig);
98  LPTMR_Init(LPTMR0, &lptmrConfig);
99  LPTMR_SetTimerPeriod(LPTMR0, MSEC_TO_COUNT(MPXV5004DR_ODR_ms, CLOCK_GetFreq(kCLOCK_LpoClk)));
100  LPTMR_EnableInterrupts(LPTMR0, kLPTMR_TimerInterruptEnable);
101  EnableIRQ(PWT_LPTMR0_IRQn);
102  LPTMR_StartTimer(LPTMR0);
103 
104  /* Configure ADC. */
105  /* Set ADC12's clock source to be Slow IRC async clock. */
106  CLOCK_SetIpSrc(kCLOCK_Adc0, kCLOCK_IpSrcSircAsync);
107  EnableIRQ(ADC0_IRQn);
108  ADC12_GetDefaultConfig(&adc12ConfigStruct);
109  adc12ConfigStruct.clockSource = kADC12_ClockSourceAlt0;
110  adc12ConfigStruct.resolution = kADC12_Resolution12Bit;
111  ADC12_Init(ADC0, &adc12ConfigStruct);
112  /* Set to software trigger mode. */
113  ADC12_EnableHardwareTrigger(ADC0, false);
114  /* Calibrate ADC. */
115  if (kStatus_Success != ADC12_DoAutoCalibration(ADC0))
116  {
117  PRINTF("ADC calibration failed!\r\n");
118  }
119  gAdcChannelConfigStruct.channelNumber = MPXVDP5004_ADC12_CHANNEL;
120  /* Enable the interrupt. */
121  gAdcChannelConfigStruct.enableInterruptOnConversionCompleted = true;
122 }

124 /*!
125  * @brief Main function
126  */
127 int main(void)
128 {
129  uint16_t pressurePascals;
130  GENERIC_DRIVER_GPIO *gpioDriver = &Driver_GPIO_KSDK;
131 
132  BOARD_InitPins();
133  BOARD_BootClockRUN();
134  BOARD_InitDebugConsole();
135 
136  PRINTF("\r\n MPXV5004DP ADC based Interrupt Example\r\n");
137 
138  /*! Initialize GREEN LED pin used by FRDM board */
139  gpioDriver->pin_init(&GREEN_LED, GPIO_DIRECTION_OUT, NULL, NULL, NULL);
140 
141  /* Initialize LPTMR and ADC framework */
142  KE15Z_LPTMR0_ADC0_Initialize();
143 
144  /* Forever loop */
145  while (true)
146  {
147  if(false == bAdcCompletedFlag)
148  {
149  SMC_SetPowerModeWait(SMC);
150  continue;
151  }
152  else
153  {
154  bAdcCompletedFlag = false;
155  gpioDriver->toggle_pin(&GREEN_LED);
156  }
157 
158  pressurePascals = (uint16_t) MPXV5004DP_PRESSURE_FROM_ADC_VALUE(gAdcConversionValue);
159  PRINTF("\r\n Differential Pressure Value: %dPa\r\n", pressurePascals);
160  }
161 }
