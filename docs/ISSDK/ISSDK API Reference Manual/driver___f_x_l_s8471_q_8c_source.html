136 int8_t FXLS8471Q_Init(struct PhysicalSensor *sensor, SensorFusionGlobals *sfg)
137 {
138  GENERIC_DRIVER_GPIO *pGPIODriver = &Driver_GPIO_KSDK;
139  int32_t status;
140  uint8_t reg;
141  if (sensor->addr == SPI_ADDR) {
142  // Initialize the Slave Select Pin.
143  pGPIODriver->pin_init(&FXLS8471_SPI_CS, GPIO_DIRECTION_OUT, NULL, NULL, NULL);
144  // Initialize SPI parameters used by ISSDK SPI routines
145  sensor->slaveParams.pReadPreprocessFN = FXLS8471Q_SPI_ReadPreprocess;
146  sensor->slaveParams.pWritePreprocessFN = FXLS8471Q_SPI_WritePreprocess;
147  sensor->slaveParams.pTargetSlavePinID = &FXLS8471_SPI_CS;
148  sensor->slaveParams.spiCmdLen = FXLS8471Q_SPI_CMD_LEN;
149  sensor->slaveParams.ssActiveValue = FXLS8471Q_SS_ACTIVE_VALUE;
150  // Read the WhoAmI
151  status = Register_SPI_Read(sensor->bus_driver, &sensor->deviceInfo, &(sensor->slaveParams), FXLS8471Q_WHO_AM_I, 1, &reg);
152  } else {
153  status = Register_I2C_Read(sensor->bus_driver, &sensor->deviceInfo, sensor->addr, FXLS8471Q_WHO_AM_I, 1, &reg);
154  }
155  if (status==SENSOR_ERROR_NONE) {
156  sfg->Accel.iWhoAmI = reg;
157  if (reg!=FXLS8471Q_WHO_AM_I_WHOAMI_VALUE) return(SENSOR_ERROR_INIT);
158  } else {
159  return(status);
160  }
161 
162  // Configure and start the FXLS8471Q sensor. This does multiple register writes
163  if (sensor->addr == SPI_ADDR)
164  status = Sensor_SPI_Write(sensor->bus_driver, &sensor->deviceInfo, &(sensor->slaveParams), FXLS8471Q_Initialization );
165  else
166  status = Sensor_I2C_Write(sensor->bus_driver, &sensor->deviceInfo, sensor->addr, FXLS8471Q_Initialization );
167 
168  // Stash some needed constants in the SF data structure for this sensor
169  sfg->Accel.iCountsPerg = FXLS8471Q_COUNTSPERG;
170  sfg->Accel.fgPerCount = 1.0F / FXLS8471Q_COUNTSPERG;
171 
172  sensor->isInitialized = F_USING_ACCEL;
173  sfg->Accel.isEnabled = true;
174 
175  return (status);
176 }

178 int8_t FXLS8471Q_Read(struct PhysicalSensor *sensor, SensorFusionGlobals *sfg)
179 {
180  uint8_t Data_Buffer[6 * FXLS8471Q_ACCEL_FIFO_SIZE]; // I2C read buffer
181  int8_t status=0; // I2C transaction status
182  int8_t j; // scratch
183  uint16_t sensor_fifo_count = 1;
184  int16_t sample[3];
185 
186  if(sensor->isInitialized != F_USING_ACCEL)
187  {
188  return SENSOR_ERROR_INIT;
189  }
190 
191  // read the F_STATUS register (mapped to STATUS) and extract number of measurements available (lower 6 bits)
192  if (sensor->addr == SPI_ADDR)
193  status = Sensor_SPI_Read(sensor->bus_driver, &sensor->deviceInfo, &(sensor->slaveParams), FXLS8471Q_F_STATUS_READ, Data_Buffer );
194  else
195  status = Sensor_I2C_Read(sensor->bus_driver, &sensor->deviceInfo, sensor->addr, FXLS8471Q_F_STATUS_READ, Data_Buffer );
196 
197  sensor_fifo_count = Data_Buffer[0] & 0x3F;
198  if (status==SENSOR_ERROR_NONE) {
199  // return if there are no measurements in the sensor FIFO.
200  // this will only occur when the FAST_LOOP_HZ equals or exceeds ACCEL_ODR_HZ
201  if (sensor_fifo_count == 0) return SENSOR_ERROR_READ;
202  } else {
203  return(status);
204  }
205 
206  FXLS8471Q_DATA_READ->numBytes = 6 * sensor_fifo_count;
207  if (sensor->addr == SPI_ADDR)
208  status = Sensor_SPI_Read(sensor->bus_driver, &sensor->deviceInfo, &sensor->slaveParams, FXLS8471Q_DATA_READ, &(Data_Buffer[0]) );
209  else
210  status = Sensor_I2C_Read(sensor->bus_driver, &sensor->deviceInfo, sensor->addr, FXLS8471Q_DATA_READ, &(Data_Buffer[0]) );
211 
212  if (status==SENSOR_ERROR_NONE) {
213  for (j=0; j<sensor_fifo_count; j++) {
214  sample[CHX] = (Data_Buffer[6*j ] << 8) | Data_Buffer[6*j + 1] ;
215  sample[CHY] = (Data_Buffer[6*j + 2] << 8) | Data_Buffer[6*j + 3] ;
216  sample[CHZ] = (Data_Buffer[6*j + 4] << 8) | Data_Buffer[6*j + 5] ;
217  conditionSample(sample); // truncate negative values to -32767
218  addToFifo((union FifoSensor*) &(sfg->Accel), ACCEL_FIFO_SIZE, sample);
219  }
220  }
221 
222  return (status);
223 }

236 {
