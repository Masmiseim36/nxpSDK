63 // Each entry in a RegisterWriteList is composed of: register address, value to write, bit-mask to apply to write (0 enables)
64 const registerwritelist_t MAG3110_Initialization[] =
65 {
66  // write 0000 0000 = 0x00 to CTRL_REG1 to place MAG3110 into standby
67  // [7-1] = 0000 000
68  // [0]: AC=0 for standby
69  { MAG3110_CTRL_REG1, 0x00, 0x00 },
70 
71  // write 1001 0000 = 0x90 to CTRL_REG2
72  // [7]: AUTO_MRST_EN=1: enable degaussing
73  // [6]: unused=0
74  // [5]: RAW=0: normal mode
75  // [4]: Mag_RST=1: enable a single degauss
76  // [3-0]: unused=0
77  { MAG3110_CTRL_REG2, 0x90, 0x00 },
78 
79  // write 000X X001 to CTRL_REG1 to set ODR and take MAG3110 out of standby
80  // [7-5]: DR=000 for 1280Hz ADC (to give best noise performance)
81  // [4-3]: OS=11 for 10Hz ODR giving 0x19
82  // [4-3]: OS=10 for 20Hz ODR giving 0x11
83  // [4-3]: OS=01 for 40Hz ODR giving 0x09
84  // [4-3]: OS=00 for 80Hz ODR giving 0x01
85  // [2]: FT=0 for normal reads
86  // [1]: TM=0 to not trigger immediate measurement
87  // [0]: AC=1 for active mode
88 #if (MAG_ODR_HZ <= 10) // select 10Hz ODR
89  { MAG3110_CTRL_REG1, 0x19, 0x00 },
90 #elif (MAG_ODR_HZ <= 30) // select 20Hz ODR (to give lower noise with standard 25Hz build)
91  { MAG3110_CTRL_REG1, 0x11, 0x00 },
92 #elif (MAG_ODR_HZ <= 40) // select 40Hz ODR
93  { MAG3110_CTRL_REG1, 0x09, 0x00 },
94 #else // select 80Hz ODR
95  { MAG3110_CTRL_REG1, 0x01, 0x00 },
96 #endif
97  __END_WRITE_DATA__
98 };

100 // All sensor drivers and initialization functions have the same prototype.
101 // sfg is a pointer to the master "global" sensor fusion structure.
102 // sensor = pointer to linked list element used by the sensor fusion subsystem to specify required sensors
103 //#pragma optimize=no_scheduling
104 int8_t MAG3110_Init(struct PhysicalSensor *sensor, SensorFusionGlobals *sfg)
105 {
106  int32_t status;
107  uint8_t reg;
108  status = Register_I2C_Read(sensor->bus_driver, &sensor->deviceInfo, sensor->addr, MAG3110_WHO_AM_I, 1, &reg);
109  if (status==SENSOR_ERROR_NONE) {
110  sfg->Mag.iWhoAmI = reg;
111  if (reg!=MAG3110_WHOAMI_VALUE) {
112  return(SENSOR_ERROR_INIT);
113  }
114  } else {
115  // iWhoAmI will return default value of zero
116  // return with error
117  return(SENSOR_ERROR_INIT);
118  }
119 
120  // Configure and start the MAG3110 sensor. This does multiple register writes
121  // (see MAG3110_Initialization definition above)
122  status = Sensor_I2C_Write(sensor->bus_driver, &sensor->deviceInfo, sensor->addr, MAG3110_Initialization );
123 
124  // Stash some needed constants in the SF data structure for this sensor
125  sfg->Mag.iCountsPeruT = MAG3110_COUNTSPERUT;
126  sfg->Mag.fCountsPeruT = (float)MAG3110_COUNTSPERUT; // IAR optimized this out without the #pragma before the function
127  sfg->Mag.fuTPerCount = 1.0F / MAG3110_COUNTSPERUT; // IAR optimized this out without the #pragma before the function
128 
129  sensor->isInitialized = F_USING_MAG; // IAR optimized this out without the #pragma before the function
130  sfg->Mag.isEnabled = true; // IAR optimized this out without the #pragma before the function
131 
132  return (status);
133 }

135 int8_t MAG3110_Read(struct PhysicalSensor *sensor, SensorFusionGlobals *sfg)
136 {
137  uint8_t I2C_Buffer[6]; // I2C read buffer
138  int8_t status; // I2C transaction status
139  int16_t sample[3]; // Reconstructed sample
140 
141  if(sensor->isInitialized != F_USING_MAG)
142  {
143  return SENSOR_ERROR_INIT;
144  }
145 
146  status = Sensor_I2C_Read(sensor->bus_driver, &sensor->deviceInfo, sensor->addr, MAG3110_DATA_READ, I2C_Buffer );
147  sample[CHX] = (I2C_Buffer[0] << 8) | I2C_Buffer[1];
148  sample[CHY] = (I2C_Buffer[2] << 8) | I2C_Buffer[3];
149  sample[CHZ] = (I2C_Buffer[4] << 8) | I2C_Buffer[5];
150  if (status==SENSOR_ERROR_NONE) {
151  conditionSample(sample); // truncate negative values to -32767
152  sample[CHZ] = -sample[CHZ]; // +Z should point up (MAG3110 Z positive is down)
153  addToFifo((union FifoSensor*) &(sfg->Mag), MAG_FIFO_SIZE, sample);
154  }
155 
156  return (status);
157 }

168 // MAG3110_Idle places the sensor into STANDBY mode (wakeup time = 25ms at ODR = 80Hz)
169 int8_t MAG3110_Idle(struct PhysicalSensor *sensor, SensorFusionGlobals *sfg)
170 {
171  int32_t status;
172  if(sensor->isInitialized == F_USING_MAG) {
173  status = Sensor_I2C_Write(sensor->bus_driver, &sensor->deviceInfo, sensor->addr, MAG3110_IDLE );
174  sensor->isInitialized = 0;
175  sfg->Mag.isEnabled = false;
176  } else {
177  return SENSOR_ERROR_INIT;
178  }
179  return status;
