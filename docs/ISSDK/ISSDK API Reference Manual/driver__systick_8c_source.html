38 // SysTick register definitions based on CMSIS definitions for same
39 #define SYST_CSR SysTick->CTRL // SysTick Control & Status Register
40 #define SYST_RVR SysTick->LOAD // SysTick Reload Value Register
41 #define SYST_CVR SysTick->VAL // SysTick Current Value Register
42 
43 // ARM-core specific function that enables the ARM systick timer on Kinetis uCs.
44 // this is purely for algorithm benchmarking and can either be deleted or re-written for use on other uCs.
45 // the systick clock frequency for each uC are defined as CORE_SYSTICK_HZ in types.h.
46 
47 // the timer is 24 bit so allows measurement of intervals up to 2^24/CORE_SYSTICK_HZ secs=0.35s for a 48MHz uC.
48 void ARM_systick_enable(void)
49 {
50  SYST_CSR = 0x5u; // enable systick from internal clock
51  SYST_RVR = 0x00FFFFFFu; // set reload to maximum 24 bit value
52  return;
53 }
54 
55 // ARM-core specific function to store the current systick timer ticks for benchmarking
56 void ARM_systick_start_ticks(int32 *pstart)
57 {
58  // store the 24 bit systick timer
59  *pstart = SYST_CVR & 0x00FFFFFF;
60 
61  return;
62 }
63 
64 // ARM-core specific function to compute the elapsed systick timer ticks for benchmarking
65 int32 ARM_systick_elapsed_ticks(int32 start_ticks)
66 {
67  int32 elapsed_ticks;
68 
69  // subtract the stored start ticks and check for wraparound down through zero
70  elapsed_ticks = start_ticks - (SYST_CVR & 0x00FFFFFF);
71  if (elapsed_ticks < 0) elapsed_ticks += SYST_RVR;
72 
73  return elapsed_ticks;
74 }

76 void ARM_systick_delay_ms(uint32 iSystemCoreClock, uint32 delay_ms)
77 {
78  int32 istart_ticks; // start ticks on entry
79  int32 ielapsed_ticks; // elapsed ticks
80  int16 i; // loop counter
81 
82  // loop for requested number of ms
83  for (i = 0; i < delay_ms; i++)
84  {
85  // loop until 1ms has elapsed
86  ARM_systick_start_ticks(&istart_ticks);
87  do
88  {
89  ielapsed_ticks = ARM_systick_elapsed_ticks(istart_ticks);
90  } while (ielapsed_ticks < iSystemCoreClock / 1000);
91  }
92 
93  return;
94 }
