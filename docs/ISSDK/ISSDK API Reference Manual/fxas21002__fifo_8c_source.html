56 /*! @def FIFO_SIZE
57  * @brief The watermark value configured for FXAS21002 FIFO Buffer.
58  */
59 #define FIFO_SIZE 4
60 
61 /*******************************************************************************
62  * Constants
63  ******************************************************************************/
64 /*! Prepare the register write list to configure FXAS21002 in FIFO mode. */
65 const registerwritelist_t fxas21002_Config_with_Fifo[] = {
66  /*! Configure CTRL_REG1 register to put FXAS21002 to 12.5Hz sampling rate. */
67  {FXAS21002_CTRL_REG1, FXAS21002_CTRL_REG1_DR_12_5HZ, FXAS21002_CTRL_REG1_DR_MASK},
68  /*! Configure F_SETUP register to set FIFO in Stop mode, Watermark of 16. */
69  {FXAS21002_F_SETUP, FXAS21002_F_SETUP_F_MODE_STOP_MODE | (FIFO_SIZE << FXAS21002_F_SETUP_F_WMRK_SHIFT),
70  FXAS21002_F_SETUP_F_MODE_MASK | FXAS21002_F_SETUP_F_WMRK_MASK},
71  {FXAS21002_CTRL_REG3, FXAS21002_CTRL_REG3_WRAPTOONE_ROLL_DATA, FXAS21002_CTRL_REG3_WRAPTOONE_MASK},
72  __END_WRITE_DATA__};
73 
74 /*! Prepare the register read list to read FXAS21002 FIFO event. */
75 const registerreadlist_t fxas21002_Fifo_Event[] = {{.readFrom = FXAS21002_F_EVENT, .numBytes = 1}, __END_READ_DATA__};
76 
77 /*! Prepare the register read list to read the raw gyro data from the FXAS21002. */
78 const registerreadlist_t fxas21002_Output_Values[] = {
79  {.readFrom = FXAS21002_OUT_X_MSB, .numBytes = FXAS21002_GYRO_DATA_SIZE * FIFO_SIZE}, __END_READ_DATA__};
80 
81 /*******************************************************************************
82  * Code
83  ******************************************************************************/
84 /*!
85  * @brief Main function
86  */
87 int main(void)
88 {
89  int32_t status;
90  uint8_t fifoEvent;
91  uint8_t data[FXAS21002_GYRO_DATA_SIZE * FIFO_SIZE];
92  fxas21002_gyrodata_t rawData;
93 
94  ARM_DRIVER_I2C *I2Cdrv = &I2C_S_DRIVER; // Now using the shield.h value!!!
95  fxas21002_i2c_sensorhandle_t FXAS21002drv;
96 
97  /*! Initialize the MCU hardware. */
98  BOARD_InitPins();
99  BOARD_BootClockRUN();
100  BOARD_SystickEnable();
101  BOARD_InitDebugConsole();
102 
103  PRINTF("\r\n ISSDK FXAS21002 sensor driver example demonstration with fifo mode\r\n");
104 
105  /*! Initialize the I2C driver. */
106  status = I2Cdrv->Initialize(I2C_S_SIGNAL_EVENT);
107  if (ARM_DRIVER_OK != status)
108  {
109  PRINTF("\r\n I2C Initialization Failed\r\n");
110  return -1;
111  }
112 
113  /*! Set the I2C Power mode. */
114  status = I2Cdrv->PowerControl(ARM_POWER_FULL);
115  if (ARM_DRIVER_OK != status)
116  {
117  PRINTF("\r\n I2C Power Mode setting Failed\r\n");
118  return -1;
119  }
120 
121  /*! Set the I2C bus speed. */
122  status = I2Cdrv->Control(ARM_I2C_BUS_SPEED, ARM_I2C_BUS_SPEED_FAST);
123  if (ARM_DRIVER_OK != status)
124  {
125  PRINTF("\r\n I2C Control Mode setting Failed\r\n");
126  return -1;
127  }
128 
129  /*! Initialize the FXAS21002 sensor driver. */
130  status = FXAS21002_I2C_Initialize(&FXAS21002drv, &I2C_S_DRIVER, I2C_S_DEVICE_INDEX, FXAS21002_I2C_ADDR,
131  FXAS21002_WHO_AM_I_WHOAMI_PROD_VALUE);
132  if (SENSOR_ERROR_NONE != status)
133  {
134  PRINTF("\r\n Sensor Initialization Failed\r\n");
135  return -1;
136  }
137  PRINTF("\r\n Successfully Initiliazed Sensor\r\n");
138 
139  /*! Set the task to be executed while waiting for I2C transactions to complete. */
140  FXAS21002_I2C_SetIdleTask(&FXAS21002drv, (registeridlefunction_t)SMC_SetPowerModeWait, SMC);
141 
142  /*! Configure the FXAS21002 sensor driver. */
143  status = FXAS21002_I2C_Configure(&FXAS21002drv, fxas21002_Config_with_Fifo);
144  if (SENSOR_ERROR_NONE != status)
145  {
146  PRINTF("\r\n FXAS21002 Sensor Configuration Failed, Err = %d\r\n", status);
147  return -1;
148  }
149  PRINTF("\r\n Successfully Applied FXAS21002 Sensor Configuration\r\n");
150 
151  for (;;) /* Forever loop */
152  {
153  /*! Wait for the FIFO watermark event. */
154  status = FXAS21002_I2C_ReadData(&FXAS21002drv, fxas21002_Fifo_Event, &fifoEvent);
155  if (0 == (fifoEvent & FXAS21002_F_EVENT_F_EVENT_MASK))
156  {
157  continue;
158  }
159 
160  /*! Read the raw sensor data from the FXAS21002. */
161  status = FXAS21002_I2C_ReadData(&FXAS21002drv, fxas21002_Output_Values, data);
162  if (ARM_DRIVER_OK != status)
163  {
164  PRINTF("\r\n Read Failed. \r\n");
165  return -1;
166  }
167 
168  for (uint8_t i = 0; i < FIFO_SIZE; i++)
169  {
170  /*! Convert the raw sensor data to signed 16-bit container for display to the debug port. */
171  rawData.gyro[0] =
172  ((int16_t)data[i * FXAS21002_GYRO_DATA_SIZE + 0] << 8) | data[i * FXAS21002_GYRO_DATA_SIZE + 1];
173  rawData.gyro[1] =
174  ((int16_t)data[i * FXAS21002_GYRO_DATA_SIZE + 2] << 8) | data[i * FXAS21002_GYRO_DATA_SIZE + 3];
175  rawData.gyro[2] =
176  ((int16_t)data[i * FXAS21002_GYRO_DATA_SIZE + 4] << 8) | data[i * FXAS21002_GYRO_DATA_SIZE + 5];
177  }
178  /* NOTE: PRINTF is relatively expensive in terms of CPU time, specially when used with-in execution loop. */
179  PRINTF("\r\n Gyro X = %d Y = %d Z = %d\r\n", rawData.gyro[0], rawData.gyro[1], rawData.gyro[2]);
180  ASK_USER_TO_RESUME(100 / FIFO_SIZE); /* Ask for user input after processing 100 samples. */
181  }
182 }
