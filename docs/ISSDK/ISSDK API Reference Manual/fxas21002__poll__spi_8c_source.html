66 /*! Prepare the register write list to configure FXAS21002 in non-FIFO mode. */
67 const registerwritelist_t cFxas21002_Config_Normal[] = {
68  /*! Configure CTRL_REG1 register to put FXAS21002 to 12.5Hz sampling rate. */

72 /*! Prepare the register read list to read FXAS21002 DataReady status. */
73 const registerreadlist_t cFxas21002_DataReady[] = {{.readFrom = FXAS21002_STATUS, .numBytes = 1}, __END_READ_DATA__};
74 
75 /*! Prepare the register read list to read the raw gyro data from the FXAS21002. */
76 const registerreadlist_t cFxas21002_Output_Values[] = {
77  {.readFrom = FXAS21002_OUT_X_MSB, .numBytes = FXAS21002_GYRO_DATA_SIZE}, __END_READ_DATA__};

82 /*! -----------------------------------------------------------------------
83  * @brief This is the The main function implementation.
84  * @details This function invokes board initializes routines, then then brings up the sensor and
85  * finally enters an endless loop to continuously read available samples.
86  * @param[in] void This is no input parameter.
87  * @return void There is no return value.
88  * @constraints None
89  * @reeentrant No
90  * -----------------------------------------------------------------------*/
91 int main(void)
92 {
93  int32_t status;
94  uint8_t gFxas21002DataReady;
95  uint8_t data[FXAS21002_GYRO_DATA_SIZE];
96  fxas21002_gyrodata_t rawData;
97 
98  ARM_DRIVER_SPI *pSPIdriver = &SPI_S_DRIVER;
99  fxas21002_spi_sensorhandle_t fxas21002Driver;
100 
101  /*! Initialize the MCU hardware. */
102  BOARD_InitPins();
103  BOARD_BootClockRUN();
104  BOARD_SystickEnable();
105  BOARD_InitDebugConsole();
106 
107  PRINTF("\r\n ISSDK FXAS21002 sensor driver example demonstration for SPI with Poll Mode.\r\n");
108 
109  /*! Initialize the SPI driver. */
110  status = pSPIdriver->Initialize(SPI_S_SIGNAL_EVENT);
111  if (ARM_DRIVER_OK != status)
112  {
113  PRINTF("\r\n SPI Initialization Failed\r\n");
114  return -1;
115  }
116 
117  /*! Set the SPI Power mode. */
118  status = pSPIdriver->PowerControl(ARM_POWER_FULL);
119  if (ARM_DRIVER_OK != status)
120  {
121  PRINTF("\r\n SPI Power Mode setting Failed\r\n");
122  return -1;
123  }
124 
125  /*! Set the SPI Slave speed. */
126  status = pSPIdriver->Control(ARM_SPI_MODE_MASTER | ARM_SPI_CPOL1_CPHA0, SPI_S_BAUDRATE);
127  if (ARM_DRIVER_OK != status)
128  {
129  PRINTF("\r\n SPI Control Mode setting Failed\r\n");
130  return -1;
131  }
132 
133  /*! Initialize the FXAS21002 sensor driver. */
134  status = FXAS21002_SPI_Initialize(&fxas21002Driver, &SPI_S_DRIVER, SPI_S_DEVICE_INDEX, &FXAS21002_CS,
135  FXAS21002_WHO_AM_I_WHOAMI_PROD_VALUE);
136  if (SENSOR_ERROR_NONE != status)
137  {
138  PRINTF("\r\n FXAS21002 Sensor Initialization Failed\r\n");
139  return -1;
140  }
141  PRINTF("\r\n Successfully Initiliazed FXAS21002 Sensor\r\n");
142 
143  /*! Set the task to be executed while waiting for SPI transactions to complete. */
144  FXAS21002_SPI_SetIdleTask(&fxas21002Driver, (registeridlefunction_t)SMC_SetPowerModeWait, SMC);
145 
146  /*! Configure the FXAS21002 sensor driver. */
147  status = FXAS21002_SPI_Configure(&fxas21002Driver, cFxas21002_Config_Normal);
148  if (SENSOR_ERROR_NONE != status)
149  {
150  PRINTF("\r\n FXAS21002 Sensor Configuration Failed, Err = %d\r\n", status);
151  return -1;
152  }
153  PRINTF("\r\n Successfully Applied FXAS21002 Sensor Configuration\r\n");
154 
155  for (;;) /* Forever loop */
156  {
157  /*! Wait for data ready from the FXAS21002. */
158  status = FXAS21002_SPI_ReadData(&fxas21002Driver, cFxas21002_DataReady, &gFxas21002DataReady);
159  if (0 == (gFxas21002DataReady & FXAS21002_DR_STATUS_ZYXDR_MASK))
160  { /* Loop, if new sample is not available. */
161  continue;
162  }
163 
164  /*! Read the raw sensor data from the FXAS21002. */
165  status = FXAS21002_SPI_ReadData(&fxas21002Driver, cFxas21002_Output_Values, data);
166  if (ARM_DRIVER_OK != status)
167  {
168  PRINTF("\r\nRead Failed.\r\n");
169  return -1;
170  }
171 
172  /*! Convert the raw sensor data for display to the debug port. */
173  rawData.gyro[0] = ((int16_t)data[0] << 8) | data[1];
174  rawData.gyro[1] = ((int16_t)data[2] << 8) | data[3];
175  rawData.gyro[2] = ((int16_t)data[4] << 8) | data[5];
176 
177  /* NOTE: PRINTF is relatively expensive in terms of CPU time, specially when used with-in execution loop. */
178  PRINTF("\r\n Gyro X = %d Y = %d Z = %d \r\n", rawData.gyro[0], rawData.gyro[1], rawData.gyro[2]);
179  }
180 }
