58 void FXLC95000_SPI_ReadPreprocess(void *pCmdOut, uint32_t offset, uint32_t size)
59 {
60  spiCmdParams_t *pSlaveCmd = pCmdOut;
61 
62  uint8_t *pWBuff = fxlc95000_spiRead_CmdBuffer;
63  uint8_t *pRBuff = fxlc95000_spiRead_DataBuffer;
64 
65  /* Formatting for Read command of FXLC95000 SENSOR. */
66  *(pWBuff) =
67  (offset & 0x7F) << 1; /* offset is the internal register address of the sensor at which write is performed. */
68 
69  /* Create the slave read command. */
70  pSlaveCmd->size = size + FXLC95000_SPI_CMD_LEN;
71  pSlaveCmd->pWriteBuffer = pWBuff;
72  pSlaveCmd->pReadBuffer = pRBuff;
73 }

75 void FXLC95000_SPI_WritePreprocess(void *pCmdOut, uint32_t offset, uint32_t size, void *pWritebuffer)
76 {
77  spiCmdParams_t *pSlaveCmd = pCmdOut;
78 
79  uint8_t *pWBuff = fxlc95000_spiWrite_CmdDataBuffer;
80  uint8_t *pRBuff = fxlc95000_spiWrite_CmdDataBuffer + size + FXLC95000_SPI_CMD_LEN;
81 
82  /* Formatting for Write command of FXLC95000 SENSOR. */
83  *(pWBuff) =
84  (offset | 0x40) << 1; /* offset is the internal register address of the sensor at which write is performed. */
85 
86  /* Copy the slave write command */
87  memcpy(pWBuff + FXLC95000_SPI_CMD_LEN, pWritebuffer, size);
88 
89  /* Create the slave command. */
90  pSlaveCmd->size = size + FXLC95000_SPI_CMD_LEN;
91  pSlaveCmd->pWriteBuffer = pWBuff;
92  pSlaveCmd->pReadBuffer = pRBuff;
93 }

95 int32_t FXLC95000_SPI_Initialize(fxlc95000_spi_sensorhandle_t *pSensorHandle,
96  ARM_DRIVER_SPI *pBus,
97  uint8_t index,
98  void *pSpiSelect,
99  void *pSlaveSelect,
100  void *pReset,
101  uint16_t buildId)
102 {
103  int32_t status;
104  uint16_t readNumber[2];
105  GENERIC_DRIVER_GPIO *pGPIODriver = &Driver_GPIO_KSDK;
106 
107  /*! Check the input parameters. */
108  if (pSensorHandle == NULL || pBus == NULL || pSpiSelect == NULL || pSlaveSelect == NULL || pReset == NULL)
109  {
110  return SENSOR_ERROR_INVALID_PARAM;
111  }
112 
113  /*! Initialize the sensor handle. */
114  pSensorHandle->pCommDrv = pBus;
115  pSensorHandle->slaveParams.pReadPreprocessFN = FXLC95000_SPI_ReadPreprocess;
116  pSensorHandle->slaveParams.pWritePreprocessFN = FXLC95000_SPI_WritePreprocess;
117  pSensorHandle->slaveParams.pTargetSlavePinID = pSlaveSelect;
118  pSensorHandle->slaveParams.spiCmdLen = FXLC95000_SPI_CMD_LEN;
119  pSensorHandle->slaveParams.ssActiveValue = FXLC95000_SS_ACTIVE_VALUE;
120 
121  pSensorHandle->deviceInfo.deviceInstance = index;
122  pSensorHandle->deviceInfo.functionParam = NULL;
123  pSensorHandle->deviceInfo.idleFunction = NULL;
124 
125  /* Initialize the Slave Select and reset Pins. */
126  pGPIODriver->pin_init(pSlaveSelect, GPIO_DIRECTION_OUT, NULL, NULL, NULL);
127  pGPIODriver->pin_init(pSpiSelect, GPIO_DIRECTION_OUT, NULL, NULL, NULL);
128  pGPIODriver->pin_init(pReset, GPIO_DIRECTION_OUT, NULL, NULL, NULL);
129 
130  /* Pull down SPI Select, Pull up SS and Reset */
131  pGPIODriver->set_pin(pSlaveSelect);
132  pGPIODriver->clr_pin(pSpiSelect);
133  pGPIODriver->set_pin(pReset);
134  BOARD_DELAY_ms(30); /* Wait for Part to Complete Reset */
135 
136  pGPIODriver->clr_pin(pReset);
137  BOARD_DELAY_ms(50); /* Wait for Part to Initialize in SPI Mode */
138 
139  pGPIODriver->set_pin(pSpiSelect);
140  BOARD_DELAY_ms(10); /* Wait for ROM Initialization before communication */
141 
142  /* Verify if the SPI connection by writing command to Boot to flash and fetch Device Info.
143  * A matching Build Number confirms Sensor Identity and Firmware. */
144  Register_SPI_BlockWrite(pSensorHandle->pCommDrv, &pSensorHandle->deviceInfo, &pSensorHandle->slaveParams, 0,
145  BootToFlash, sizeof(BootToFlash));
146  BOARD_DELAY_ms(10); /* Allow time for the device to boot to Flash. */
147  Register_SPI_BlockWrite(pSensorHandle->pCommDrv, &pSensorHandle->deviceInfo, &pSensorHandle->slaveParams, 0,
148  GetDeviceInfoCmd, sizeof(GetDeviceInfoCmd));
149  BOARD_DELAY_ms(1);
150 
151  status = Register_SPI_Read(pSensorHandle->pCommDrv, &pSensorHandle->deviceInfo, &pSensorHandle->slaveParams,
152  FXLC95000_BUILD_ID_OFFSET, FXLC95000_BUILD_ID_SIZE + FXLC95000_PART_NUMBER_SIZE,
153  (uint8_t *)&readNumber);
154  readNumber[0] = (readNumber[0] >> 8) | (readNumber[0] << 8);
155  readNumber[1] = (readNumber[1] >> 8) | (readNumber[1] << 8);
156  if (ARM_DRIVER_OK != status || readNumber[0] != buildId || readNumber[1] != FXLC95000_PART_NUMBER)
157  {
158  pSensorHandle->isInitialized = false;
159  return SENSOR_ERROR_INIT;
160  }
161 
162  pSensorHandle->isInitialized = true;
163  return SENSOR_ERROR_NONE;
164 }

198  {
199  return SENSOR_ERROR_WRITE;
200  }
201  }
202

205  {
206  status = Sensor_SPI_Read(pSensorHandle->pCommDrv, &pSensorHandle->deviceInfo, &pSensorHandle->slaveParams,
207  pResponseList, pBuffer);
208  if (ARM_DRIVER_OK != status)
209  {
210  return SENSOR_ERROR_READ;
211  }
212  }
213 
214  return SENSOR_ERROR_NONE;

262 int32_t FXLC95000_I2C_FlashPayload(
263  ARM_DRIVER_I2C *pBus, uint8_t index, uint16_t sAddress, uint8_t *pFlashBytes, uint8_t numBytes)
264 {
265  int32_t status;
266 
267  /* First 4 bytes is address (which fits into 2 byte separate address field and Bank code)
268  * and then remaining bytes is WData. */
269  /* Make sure that the Flash payload received has the minimum information bytes
270  * required to construct a CI_READ_WRITE ROM Command. */
271  if (numBytes >= FXLC95000_FLASH_PAYLOAD_ADDR_LEN + FXLC95000_ROM_CI_WRITE_MIN_LEN)
272  {
273  uint8_t bytesToPageBoundary = 256 - *(pFlashBytes + 3);
274  uint8_t bytesAfterPageBoundary = numBytes - bytesToPageBoundary - FXLC95000_FLASH_PAYLOAD_ADDR_LEN;
275  /* Prepare CI_READ_WRITE ROM Command. */
276  /* Check if the write command being attempted spans across a page boundary (256 bytes). */
277  if (bytesToPageBoundary > 0 && bytesToPageBoundary < numBytes - FXLC95000_FLASH_PAYLOAD_ADDR_LEN)
278  { /* Break it into 2 write commands if write spans across 2 pages. Ref. Section 16.5 of FXLC95000CLHWRM. */
279  uint8_t writeFlashCommand[FXLC95000_ROM_CI_READ_WRITE_MAX_LEN] = {0x0A, 0x00, 0x12, 0x34, 0x56, 0x78};
280  uint8_t writeFlashCommand_2[FXLC95000_ROM_CI_READ_WRITE_MAX_LEN] = {0x0A, 0x00, 0x12, 0x34, 0x56, 0x78};
281 
282  writeFlashCommand[1] =
283  (*(pFlashBytes + 1) == 0) ? 0x1F & (bytesToPageBoundary) : 0xC0 | (bytesToPageBoundary);
284  for (int i = 0; i < bytesToPageBoundary + FXLC95000_ROM_CI_READ_WRITE_ADDR_LEN;
285  i++) // 2 LSB Bytes of Address + Data.
286  {
287  writeFlashCommand[FXLC95000_ROM_CI_READ_WRITE_HDR_LEN + i] =
288  pFlashBytes[FXLC95000_ROM_CI_READ_WRITE_ADDR_LEN + i];
289  }
290 
291  writeFlashCommand_2[1] =
292  (*(pFlashBytes + 1) == 0) ? 0x1F & (bytesAfterPageBoundary) : 0xC0 | (bytesAfterPageBoundary);
293  uint8_t address[] = {*(pFlashBytes + FXLC95000_ROM_CI_READ_WRITE_ADDR_LEN + 1),
294  *(pFlashBytes + FXLC95000_ROM_CI_READ_WRITE_ADDR_LEN)};
295  *((uint16_t *)address) += bytesToPageBoundary;
296  writeFlashCommand_2[FXLC95000_ROM_CI_READ_WRITE_HDR_LEN] = address[1];
297  writeFlashCommand_2[FXLC95000_ROM_CI_READ_WRITE_HDR_LEN + 1] = address[0];
298  for (int i = 0; i < bytesAfterPageBoundary; i++)
299  {
300  writeFlashCommand_2[FXLC95000_ROM_CI_READ_WRITE_HDR_LEN + FXLC95000_ROM_CI_READ_WRITE_ADDR_LEN + i] =
301  pFlashBytes[FXLC95000_FLASH_PAYLOAD_ADDR_LEN + bytesToPageBoundary + i];
302  }
303 
304  if (bytesToPageBoundary > FXLC95000_ROM_CI_WRITE_MAX_LEN ||
305  bytesAfterPageBoundary > FXLC95000_ROM_CI_WRITE_MAX_LEN)
306  {
307  return SENSOR_ERROR_INVALID_PARAM;
308  }
309  else
310  { /* Create Command List to Write. */
311  const registercommandlist_t cWriteFlashSequence[] = {
312  {writeFlashCommand, 0,
313  FXLC95000_ROM_CI_READ_WRITE_HDR_LEN + FXLC95000_ROM_CI_READ_WRITE_ADDR_LEN + bytesToPageBoundary},
314  {writeFlashCommand_2, 0, FXLC95000_ROM_CI_READ_WRITE_HDR_LEN +
315  FXLC95000_ROM_CI_READ_WRITE_ADDR_LEN + bytesAfterPageBoundary},
316  __END_WRITE_CMD__};
317  /* Write Flash Bytes as Commands to Sensor. */
318  status = FXLC95000_I2C_FlashCommands(pBus, index, sAddress, cWriteFlashSequence);
319  if (ARM_DRIVER_OK != status)
320  {
321  return SENSOR_ERROR_WRITE;
322  }
323  }
324  }
325  else /* If write is enclosed within 1-page, a single CI_READ_WRITE will do the job. */
326  {
327  uint8_t writeFlashCommand[FXLC95000_ROM_CI_READ_WRITE_MAX_LEN] = {0x0A, 0x00, 0x12, 0x34, 0x56, 0x78};
328  writeFlashCommand[1] = /* Ref. Section 16.5 of FXLC95000CLHWRM. */
329  (*(pFlashBytes + 1) == 0) ? 0x1F & (numBytes - FXLC95000_FLASH_PAYLOAD_ADDR_LEN) :
330  0xC0 | (numBytes - FXLC95000_FLASH_PAYLOAD_ADDR_LEN);
331 
332  for (int i = FXLC95000_ROM_CI_READ_WRITE_ADDR_LEN; i < numBytes; i++) // Skip 2 MSBs of Address.
333  {
334  writeFlashCommand[FXLC95000_ROM_CI_READ_WRITE_HDR_LEN + i - FXLC95000_ROM_CI_READ_WRITE_ADDR_LEN] =
335  pFlashBytes[i];
336  }
337  if (FXLC95000_ROM_CI_READ_WRITE_HDR_LEN + numBytes >
338  FXLC95000_ROM_CI_WRITE_MAX_LEN + FXLC95000_ROM_CI_READ_WRITE_ADDR_LEN)
339  {
340  return SENSOR_ERROR_INVALID_PARAM;
341  }
342  else
343  { /* Create Command List to Write. */
344  const registercommandlist_t cWriteFlashSequence[] = {
345  {writeFlashCommand, 0,
346  FXLC95000_ROM_CI_READ_WRITE_HDR_LEN + numBytes - FXLC95000_ROM_CI_READ_WRITE_ADDR_LEN},
347  __END_WRITE_CMD__};
348  /* Write Flash Bytes as Commands to Sensor. */
349  status = FXLC95000_I2C_FlashCommands(pBus, index, sAddress, cWriteFlashSequence);
350  if (ARM_DRIVER_OK != status)
351  {
352  return SENSOR_ERROR_WRITE;
353  }
354  }
355  }
356  }
357  else
358  {
359  return SENSOR_ERROR_INVALID_PARAM;
360  }
361 
362  return SENSOR_ERROR_NONE;
363 }
