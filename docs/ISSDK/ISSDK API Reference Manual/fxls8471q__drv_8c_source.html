57 void FXLS8471Q_SPI_ReadPreprocess(void *pCmdOut, uint32_t offset, uint32_t size)
58 {
59  spiCmdParams_t *pSlaveCmd = pCmdOut;
60 
61  uint8_t *pWBuff = fxls8471q_spiRead_CmdBuffer;
62  uint8_t *pRBuff = fxls8471q_spiRead_DataBuffer;
63 
64  /* Formatting for Read command of FXLS8471Q SENSOR. */
65  *(pWBuff) = offset & 0x7F; /* offset is the internal register address of the sensor at which write is performed. */
66  *(pWBuff + 1) = offset & 0x80;
67 
68  // Create the slave read command.
69  pSlaveCmd->size = size + FXLS8471Q_SPI_CMD_LEN;
70  pSlaveCmd->pWriteBuffer = pWBuff;
71  pSlaveCmd->pReadBuffer = pRBuff;
72 }

74 void FXLS8471Q_SPI_WritePreprocess(void *pCmdOut, uint32_t offset, uint32_t size, void *pWritebuffer)
75 {
76  spiCmdParams_t *pSlaveCmd = pCmdOut;
77 
78  uint8_t *pWBuff = fxls8471q_spiWrite_CmdDataBuffer;
79  uint8_t *pRBuff = fxls8471q_spiWrite_CmdDataBuffer + size + FXLS8471Q_SPI_CMD_LEN;
80 
81  /* Formatting for Write command of FXLS8471Q SENSOR. */
82  *(pWBuff) = offset | 0x80; /* offset is the internal register address of the sensor at which write is performed. */
83  *(pWBuff + 1) = offset & 0x80;
84 
85  /* Copy the slave write command */
86  memcpy(pWBuff + FXLS8471Q_SPI_CMD_LEN, pWritebuffer, size);
87 
88  /* Create the slave command. */
89  pSlaveCmd->size = size + FXLS8471Q_SPI_CMD_LEN;
90  pSlaveCmd->pWriteBuffer = pWBuff;
91  pSlaveCmd->pReadBuffer = pRBuff;
92 }

94 int32_t FXLS8471Q_SPI_Initialize(fxls8471q_spi_sensorhandle_t *pSensorHandle,
95  ARM_DRIVER_SPI *pBus,
96  uint8_t index,
97  void *pSlaveSelect,
98  uint8_t whoAmi)
99 {
100  int32_t status;
101  uint8_t reg;
102  GENERIC_DRIVER_GPIO *pGPIODriver = &Driver_GPIO_KSDK;
103 
104  /*! Check the input parameters. */
105  if ((pSensorHandle == NULL) || (pBus == NULL) || (pSlaveSelect == NULL))
106  {
107  return SENSOR_ERROR_INVALID_PARAM;
108  }
109 
110  /*! Initialize the sensor handle. */
111  pSensorHandle->pCommDrv = pBus;
112  pSensorHandle->slaveParams.pReadPreprocessFN = FXLS8471Q_SPI_ReadPreprocess;
113  pSensorHandle->slaveParams.pWritePreprocessFN = FXLS8471Q_SPI_WritePreprocess;
114  pSensorHandle->slaveParams.pTargetSlavePinID = pSlaveSelect;
115  pSensorHandle->slaveParams.spiCmdLen = FXLS8471Q_SPI_CMD_LEN;
116  pSensorHandle->slaveParams.ssActiveValue = FXLS8471Q_SS_ACTIVE_VALUE;
117 
118  pSensorHandle->deviceInfo.deviceInstance = index;
119  pSensorHandle->deviceInfo.functionParam = NULL;
120  pSensorHandle->deviceInfo.idleFunction = NULL;
121 
122  /* Initialize the Slave Select Pin. */
123  pGPIODriver->pin_init(pSlaveSelect, GPIO_DIRECTION_OUT, NULL, NULL, NULL);
124 
125  /*! Read and store the device's WHO_AM_I.*/
126  status = Register_SPI_Read(pSensorHandle->pCommDrv, &pSensorHandle->deviceInfo, &pSensorHandle->slaveParams,
127  FXLS8471Q_WHO_AM_I, 1, &reg);
128  if ((ARM_DRIVER_OK != status) || (whoAmi != reg))
129  {
130  pSensorHandle->isInitialized = false;
131  return SENSOR_ERROR_INIT;
132  }
133 
134  pSensorHandle->isInitialized = true;
135  return SENSOR_ERROR_NONE;
136 }

146 int32_t FXLS8471Q_SPI_Configure(fxls8471q_spi_sensorhandle_t *pSensorHandle, const registerwritelist_t *pRegWriteList)
147 {
148  int32_t status;
149 
150  /*! Validate for the correct handle and register write list.*/
151  if ((pSensorHandle == NULL) || (pRegWriteList == NULL))
152  {
153  return SENSOR_ERROR_INVALID_PARAM;
154  }
155 
156  /*! Check whether sensor handle is initialized before applying configuration.*/
157  if (pSensorHandle->isInitialized != true)
158  {
159  return SENSOR_ERROR_INIT;
160  }
161 
162  /*! Put the device into standby mode so that configuration can be applied.*/
163  status = Register_SPI_Write(pSensorHandle->pCommDrv, &pSensorHandle->deviceInfo, &pSensorHandle->slaveParams,
164  FXLS8471Q_CTRL_REG1, FXLS8471Q_CTRL_REG1_MODE_STANDBY, FXLS8471Q_CTRL_REG1_MODE_MASK);
165  if (ARM_DRIVER_OK != status)
166  {
167  return SENSOR_ERROR_WRITE;
168  }
169 
170  /*! Apply the Sensor Configuration based on the Register Write List */
171  status = Sensor_SPI_Write(pSensorHandle->pCommDrv, &pSensorHandle->deviceInfo, &pSensorHandle->slaveParams,
172  pRegWriteList);
173  if (ARM_DRIVER_OK != status)
174  {
175  return SENSOR_ERROR_WRITE;
176  }
177 
178  /*! Put the device into active mode and ready for reading data.*/
179  status = Register_SPI_Write(pSensorHandle->pCommDrv, &pSensorHandle->deviceInfo, &pSensorHandle->slaveParams,
180  FXLS8471Q_CTRL_REG1, FXLS8471Q_CTRL_REG1_MODE_ACTIVE, FXLS8471Q_CTRL_REG1_MODE_MASK);
181  if (ARM_DRIVER_OK != status)
182  {
183  return SENSOR_ERROR_WRITE;
184  }
185 
186  return SENSOR_ERROR_NONE;
187 }

189 int32_t FXLS8471Q_SPI_ReadData(fxls8471q_spi_sensorhandle_t *pSensorHandle,
190  const registerreadlist_t *pReadList,
191  uint8_t *pBuffer)
192 {
193  int32_t status;
194

197  {
198  return SENSOR_ERROR_INVALID_PARAM;
199  }
200 
201  /*! Check whether sensor handle is initialized before reading sensor data.*/
202  if (pSensorHandle->isInitialized != true)
203  {
204  return SENSOR_ERROR_INIT;

206 
207  /*! Parse through the read list and read the data one by one. */
208  status = Sensor_SPI_Read(pSensorHandle->pCommDrv, &pSensorHandle->deviceInfo, &pSensorHandle->slaveParams,
209  pReadList, pBuffer);

211  {
212  return SENSOR_ERROR_READ;
213  }
214 
215  return SENSOR_ERROR_NONE;

218 int32_t FXLS8471Q_SPI_Deinit(fxls8471q_spi_sensorhandle_t *pSensorHandle)
219 {
220  if (pSensorHandle == NULL)
221  {
222  return SENSOR_ERROR_INVALID_PARAM;
223  }
224 
225  /*! Check whether sensor handle is initialized before triggering sensor reset.*/
226  if (pSensorHandle->isInitialized != true)
227  {
228  return SENSOR_ERROR_INIT;
229  }
230 
231  /* Note:
232  * There is a Errata described in the data sheet for soft reset with SPI,
233  * which says device specific paratameters do not get updated correctly from NVM,
234  * causing inaccurate data output once the soft-reset command issued in SPI mode.
235  * The work around is hard reset, which cannot be handled in sensor driver software.
236  * So software reset command wont be issued until this issue is fixed in the silicon.
237  */
238 
239  /*! De-initialize sensor handle. */
240  pSensorHandle->isInitialized = false;
241 
242  return SENSOR_ERROR_NONE;
243 }
