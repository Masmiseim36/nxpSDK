70  /*! System and Control registers. */
71  /*! Configure the FXOS8700 to 12.5Hz sampling rate. */
72  {FXOS8700_CTRL_REG1, FXOS8700_CTRL_REG1_DR_SINGLE_12P5_HZ, FXOS8700_CTRL_REG1_DR_MASK},

74  FXOS8700_CTRL_REG3_IPOL_MASK | FXOS8700_CTRL_REG3_PP_OD_MASK}, /*! Active High, Push-Pull */
75  {FXOS8700_CTRL_REG4, FXOS8700_CTRL_REG4_INT_EN_DRDY_EN,

79  {FXOS8700_M_CTRL_REG2, FXOS8700_M_CTRL_REG2_M_AUTOINC_ACCEL_ONLY_MODE, FXOS8700_M_CTRL_REG2_M_AUTOINC_MASK},

97 /*! -----------------------------------------------------------------------
98  * @brief This is the Sensor Data Ready ISR implementation.
99  * @details This function sets the flag which indicates if a new sample(s) is available for reading.
100  * @param[in] pUserData This is a void pointer to the instance of the user specific data structure for the ISR.
101  * @return void There is no return value.
102  * @constraints None
103  * @reeentrant Yes
104  * -----------------------------------------------------------------------*/
105 void fxos8700_isr_callback(void *pUserData)
106 { /*! @brief Set flag to indicate Sensor has signalled data ready. */
107  fxos8700Interrupt = true;
108 }
109 
110 /*! -----------------------------------------------------------------------
111  * @brief This is the The main function implementation.
112  * @details This function invokes board initializes routines, then then brings up the sensor and
113  * finally enters an endless loop to continuously read available samples.
114  * @param[in] void This is no input parameter.
115  * @return void There is no return value.
116  * @constraints None
117  * @reeentrant No
118  * -----------------------------------------------------------------------*/
119 int main(void)
120 {
121  int32_t status;
122  uint8_t data[RAW_ACCEL_DATA_SIZE];
123  fxos8700_accelmagdata_t rawData;
124 
125  ARM_DRIVER_I2C *I2Cdrv = &I2C_S_DRIVER; // Now using the shield.h value!!!
126  fxos8700_i2c_sensorhandle_t FXOS8700drv;
127  GENERIC_DRIVER_GPIO *gpioDriver = &Driver_GPIO_KSDK;
128 
129  /*! Initialize the MCU hardware. */
130  BOARD_InitPins();
131  BOARD_BootClockRUN();
132  BOARD_SystickEnable();
133  BOARD_InitDebugConsole();
134 
135  PRINTF("\r\n ISSDK FXOS8700 sensor driver example demonstration with interrupt mode.\r\n");
136 
137  /*! Initialize INT1_FXAS21002 pin used by FRDM board */
138  gpioDriver->pin_init(&FXOS8700_INT2, GPIO_DIRECTION_IN, NULL, &fxos8700_isr_callback, NULL);
139 
140  /*! Initialize RGB LED pin used by FRDM board */
141  gpioDriver->pin_init(&GREEN_LED, GPIO_DIRECTION_OUT, NULL, NULL, NULL);
142 
143  /*! Initialize the I2C driver. */
144  status = I2Cdrv->Initialize(I2C_S_SIGNAL_EVENT);
145  if (ARM_DRIVER_OK != status)
146  {
147  PRINTF("\r\n I2C Initialization Failed\r\n");
148  return -1;
149  }
150 
151  /*! Set the I2C Power mode. */
152  status = I2Cdrv->PowerControl(ARM_POWER_FULL);
153  if (ARM_DRIVER_OK != status)
154  {
155  PRINTF("\r\n I2C Power Mode setting Failed\r\n");
156  return -1;
157  }
158 
159  /*! Set the I2C bus speed. */
160  status = I2Cdrv->Control(ARM_I2C_BUS_SPEED, ARM_I2C_BUS_SPEED_FAST);
161  if (ARM_DRIVER_OK != status)
162  {
163  PRINTF("\r\n I2C Control Mode setting Failed\r\n");
164  return -1;
165  }
166 
167  /*! Initialize the FXOS8700 sensor driver. */
168  status = FXOS8700_I2C_Initialize(&FXOS8700drv, &I2C_S_DRIVER, I2C_S_DEVICE_INDEX, FXOS8700_I2C_ADDR,
169  FXOS8700_WHO_AM_I_PROD_VALUE);
170  if (SENSOR_ERROR_NONE != status)
171  {
172  PRINTF("\r\n Sensor Initialization Failed\r\n");
173  return -1;
174  }
175  PRINTF("\r\n Successfully Initiliazed Sensor\r\n");
176 
177  /*! Set the task to be executed while waiting for I2C transactions to complete. */
178  FXOS8700_I2C_SetIdleTask(&FXOS8700drv, (registeridlefunction_t)SMC_SetPowerModeWait, SMC);
179 
180  /*! Configure the fxos8700 sensor driver. */
181  status = FXOS8700_I2C_Configure(&FXOS8700drv, fxos8700_Config_Interrupt);
182  if (SENSOR_ERROR_NONE != status)
183  {
184  PRINTF("\r\n FXOS8700 Sensor Configuration Failed, Err = %d\r\n", status);
185  return -1;
186  }
187  PRINTF("\r\n Successfully Applied FXOS8700 Sensor Configuration\r\n");
188 
189  for (;;) /* Forever loop */
190  { /* In ISR Mode we do not need to check Data Ready Register.
191  * The receipt of interrupt will indicate data is ready. */
192  if (false == fxos8700Interrupt)
193  { /* Loop, if new sample is not available. */
194  continue;
195  }
196  else
197  { /*! Clear the data ready flag, it will be set again by the ISR. */
198  fxos8700Interrupt = false;
199  gpioDriver->toggle_pin(&GREEN_LED);
200  }
201 
202  /*! Read the raw sensor data from the fxos8700. */
203  status = FXOS8700_I2C_ReadData(&FXOS8700drv, FXOS8700_ACCEL_READ, data);
204  if (ARM_DRIVER_OK != status)
205  {
206  PRINTF("\r\n Read Failed. \r\n");
207  return -1;
208  }
209 
210  /*! Convert the raw sensor data to signed 16-bit container for display to the debug port. */
211  rawData.accel[0] = ((int16_t)data[0] << 8) | data[1];
212  rawData.accel[0] /= 4;
213  rawData.accel[1] = ((int16_t)data[2] << 8) | data[3];
214  rawData.accel[1] /= 4;
215  rawData.accel[2] = ((int16_t)data[4] << 8) | data[5];
216  rawData.accel[2] /= 4;
217 
218  /* NOTE: PRINTF is relatively expensive in terms of CPU time, specially when used with-in execution loop. */
219  PRINTF("\r\n Accel X = %d Y = %d Z = %d\r\n", rawData.accel[0], rawData.accel[1], rawData.accel[2]);
220  ASK_USER_TO_RESUME(50); /* Ask for user input after processing 100 samples. */
221  }
222 }
