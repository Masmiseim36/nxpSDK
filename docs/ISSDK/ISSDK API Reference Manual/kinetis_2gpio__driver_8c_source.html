76 /***********************************************************************
77  *
78  * Function Name : ksdk_gpio_pin_init
79  * Description : Initialize particular GPIO pin used by board.
80  *
81  ***************************************************************************/
82 void ksdk_gpio_pin_init(
83  pinID_t aPinId, gpio_direction_t dir, void *apPinConfig, gpio_isr_handler_t aIsrHandler, void *apUserData)
84 {
85  gpioConfigKSDK_t *pGpioConfig = (gpioConfigKSDK_t *)apPinConfig;
86  gpioHandleKSDK_t *pinHandle = (gpioHandleKSDK_t *)aPinId;
87  if (NULL == apPinConfig)
88  {
89  pGpioConfig = &gpioConfigDefault;
90  pGpioConfig->portPinConfig.pullSelect = kPORT_PullUp;
91  pGpioConfig->portPinConfig.mux = kPORT_MuxAsGpio;
92  }
93  // Configure the clock
94  CLOCK_EnableClock(pinHandle->clockName);
95 
96  // Set the port pin configuration value
97  PORT_SetPinConfig(pinHandle->portBase, pinHandle->pinNumber, &pGpioConfig->portPinConfig);
98 
99  pGpioConfig->pinConfig.pinDirection = dir == GPIO_DIRECTION_IN ? kGPIO_DigitalInput : kGPIO_DigitalOutput;
100  // Set the pin information
101  GPIO_PinInit(pinHandle->base, pinHandle->pinNumber, &pGpioConfig->pinConfig);
102 
103  // Isr is installed
104  if (aIsrHandler)
105  {
106  // Enable the IRQ
107  EnableIRQ(pinHandle->irq);
108  isrObj[pinHandle->portNumber][pinHandle->pinNumber].isrHandle = aIsrHandler;
109  isrObj[pinHandle->portNumber][pinHandle->pinNumber].pUserData = apUserData;
110  // Enable the interrupt on a pin.
111  PORT_SetPinInterruptConfig(pinHandle->portBase, pinHandle->pinNumber, pGpioConfig->interruptMode);
112  }
113 }

170 /***********************************************************************
171  *
172  * Function Name : ksdk_gpio_handle_interrupt
173  * Description : handle the gpio interrupt in a pin.
174  *
175  ***************************************************************************/
176 
177 void ksdk_gpio_handle_interrupt(GPIO_Type *apBase, port_number_t aPortNumber)
178 {
179  uint32_t isfr = GPIO_GetPinsInterruptFlags(apBase);
180  // parse through all the pending interrupt for a PORT
181  for (uint8_t i = 0; i < GPIO_NUMBER_OF_PIN; i++)
182  {
183  if (isfr & (1 << i))
184  {
185  gpio_isr_handler_t handle = isrObj[aPortNumber][i].isrHandle;
186  if (handle == NULL)
187  {
188  continue;
189  }
190  handle(isrObj[aPortNumber][i].pUserData);
191  GPIO_ClearPinsInterruptFlags(apBase, (1 << i));
192  }
193  }
194 }
