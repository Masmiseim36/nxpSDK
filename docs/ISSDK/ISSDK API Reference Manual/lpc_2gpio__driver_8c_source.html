114 /***********************************************************************
115  *
116  * Function Name : ksdk_gpio_handle_interrupt
117  * Description : handle the gint interrupt of a port.
118  *
119  ***************************************************************************/
120 void issdk_gpio_handle_interrupt(GINT_Type *apBase, gint_port_t aPortNumber)
121 {
122  uint32_t enableMask, polarityMask;
123 
124  GINT_GetConfigPins(apBase, aPortNumber, &polarityMask, &enableMask);
125  // parse through all the pending interrupt for a PORT
126  for (uint8_t i = 0; i < GINT_NUMBER_OF_PIN; i++)
127  {
128  if (enableMask & (1 << i))
129  {
130  gpio_isr_handler_t handle = isrObj[aPortNumber][i].isrHandle;
131  if (handle == NULL)
132  {
133  continue;
134  }
135  // call user defined handler
136  handle(isrObj[aPortNumber][i].pUserData);
137  }
138  }
139 }

152 /***********************************************************************
153  *
154  * Function Name : ksdk_gpio_pin_init
155  * Description : Initialize particular GPIO pin used by board.
156  *
157  ***************************************************************************/
158 void ksdk_gpio_pin_init(
159  pinID_t aPinId, gpio_direction_t dir, void *apPinConfig, gpio_isr_handler_t aIsrHandler, void *apUserData)
160 {
161  uint32_t polarityMask, enableMask;
162 
163  gpioConfigKSDK_t *pGpioConfig = (gpioConfigKSDK_t *)apPinConfig;
164  gpioHandleKSDK_t *pinHandle = (gpioHandleKSDK_t *)aPinId;
165  if (NULL == apPinConfig)
166  {
167  pGpioConfig = &gpioConfigDefault;
168  }
169 
170  // Configure as GPIO
171  CLOCK_EnableClock(pinHandle->clockName);
172  pGpioConfig->pinConfig.pinDirection = (dir == GPIO_DIRECTION_IN) ? kGPIO_DigitalInput : kGPIO_DigitalOutput;
173  GPIO_PinInit(pinHandle->base, pinHandle->portNumber, pinHandle->pinNumber, &pGpioConfig->pinConfig);
174 
175  // Configure is GINT if ISR requested (There should be as many GINTs as there are GPIO Ports).
176  if (aIsrHandler && pinHandle->portNumber < GINT_COUNT)
177  {
178  /* Initialize GINT device for the associated Port */
179  GINT_Init(gintBases[pinHandle->portNumber]);
180 
181  // Enable the IRQ
182  isrObj[pinHandle->portNumber][pinHandle->pinNumber].isrHandle = aIsrHandler;
183  isrObj[pinHandle->portNumber][pinHandle->pinNumber].pUserData = apUserData;
184 
185  /* Setup GINT for edge trigger, "OR" mode */
186  GINT_SetCtrl(gintBases[pinHandle->portNumber], kGINT_CombineOr, kGINT_TrigEdge,
187  gintIsrCb[pinHandle->portNumber]);
188 
189  /* Get current pins & polarity for GINT */
190  GINT_GetConfigPins(gintBases[pinHandle->portNumber], pinHandle->portNumber, &polarityMask, &enableMask);
191 
192  /* Update for this PIN */
193  if (pGpioConfig->interruptMode == kGINT_InterruptRisingEdge)
194  {
195  polarityMask |= pinHandle->mask; /* Active High for Rising Edge */
196  }
197  else
198  {
199  polarityMask &= ~(pinHandle->mask); /* Active Low for Falling Edge */
200  }
201  enableMask |= pinHandle->mask;
202 
203  /* Select pins & polarity for GINT */
204  GINT_ConfigPins(gintBases[pinHandle->portNumber], pinHandle->portNumber, polarityMask, enableMask);
205 
206  /* Enable callbacks for GINT */
207  GINT_EnableCallback(gintBases[pinHandle->portNumber]);
208  }
209 }
