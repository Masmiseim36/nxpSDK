67 const registerwritelist_t cMag3110ConfigNormal[] = {
68  /* Set Ouput Rate @10HZ (ODR = 2 and OSR = 32). */

85 /*!
86  * @brief Main function
87  */
88 int main(void)
89 {
90  int32_t status;
91  uint8_t dataReady;
92  uint8_t data[MAG3110_DATA_SIZE];
93  mag3110_magdata_t rawData;
94 
95  ARM_DRIVER_I2C *I2Cdrv = &I2C_S_DRIVER; // Now using the shield.h value!!!
96  mag3110_i2c_sensorhandle_t mag3110Driver;
97 
98  BOARD_InitPins();
99  BOARD_BootClockRUN();
100  BOARD_SystickEnable();
101  BOARD_InitDebugConsole();
102 
103  PRINTF("\r\n ISSDK MAG3110 sensor driver example demonstration with poll mode\r\n");
104 
105  /*! Initialize the I2C driver. */
106  status = I2Cdrv->Initialize(I2C_S_SIGNAL_EVENT);
107  if (ARM_DRIVER_OK != status)
108  {
109  PRINTF("\r\n I2C Initialization Failed\r\n");
110  return -1;
111  }
112 
113  /*! Set the I2C Power mode. */
114  status = I2Cdrv->PowerControl(ARM_POWER_FULL);
115  if (ARM_DRIVER_OK != status)
116  {
117  PRINTF("\r\n I2C Power Mode setting Failed\r\n");
118  return -1;
119  }
120 
121  /*! Set the I2C bus speed. */
122  status = I2Cdrv->Control(ARM_I2C_BUS_SPEED, ARM_I2C_BUS_SPEED_FAST);
123  if (ARM_DRIVER_OK != status)
124  {
125  PRINTF("\r\n I2C Control Mode setting Failed\r\n");
126  return -1;
127  }
128 
129  /*! Initialize MAG3110 sensor driver. */
130  status = MAG3110_I2C_Initialize(&mag3110Driver, &I2C_S_DRIVER, I2C_S_DEVICE_INDEX, MAG3110_I2C_ADDR,
131  MAG3110_WHOAMI_VALUE);
132  if (SENSOR_ERROR_NONE != status)
133  {
134  PRINTF("\r\n Sensor Initialization Failed\r\n");
135  return -1;
136  }
137  PRINTF("\r\n Successfully Initiliazed Sensor\r\n");
138 
139  /*! Set the task to be executed while waiting for I2C transactions to complete. */
140  MAG3110_I2C_SetIdleTask(&mag3110Driver, (registeridlefunction_t)SMC_SetPowerModeWait, SMC);
141 
142  /*! Configure the MAG3110 sensor driver. */
143  status = MAG3110_I2C_Configure(&mag3110Driver, cMag3110ConfigNormal);
144  if (SENSOR_ERROR_NONE != status)
145  {
146  PRINTF("\r\n MAG3110 Sensor Configuration Failed, Err = %d\r\n", status);
147  return -1;
148  }
149  PRINTF("\r\n Successfully Applied MAG3110 Sensor Configuration\r\n");
150 
151  for (;;) /* Forever loop */
152  {
153  /*! Wait for data ready from the FXAS21002. */
154  status = MAG3110_I2C_ReadData(&mag3110Driver, cMag3110Status, &dataReady);
155  if (0 == (dataReady & MAG3110_DR_STATUS_ZYXDR_MASK))
156  { /* Loop, if new sample is not available. */
157  continue;
158  }
159 
160  /*! Read the raw sensor data from the FXAS21002. */
161  status = MAG3110_I2C_ReadData(&mag3110Driver, cMag3110OutputNormal, data);
162  if (ARM_DRIVER_OK != status)
163  {
164  PRINTF("\r\n Read Failed. \r\n");
165  return -1;
166  }
167 
168  /*! Process the sample and convert the raw sensor data to signed 16-bit container. */
169  rawData.mag[0] = ((int16_t)data[0] << 8) | data[1];
170  rawData.mag[1] = ((int16_t)data[2] << 8) | data[3];
171  rawData.mag[2] = ((int16_t)data[4] << 8) | data[5];
172 
173  /* NOTE: PRINTF is relatively expensive in terms of CPU time, specially when used with-in execution loop. */
174  PRINTF("\r\n Mag X = %d Y = %d Z = %d\r\n", rawData.mag[0], rawData.mag[1], rawData.mag[2]);
175  ASK_USER_TO_RESUME(100); /* Ask for user input after processing 10 samples. */
176  }
177 }
