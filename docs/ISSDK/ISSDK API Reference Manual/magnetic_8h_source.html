63 /// The Magnetometer Measurement Buffer holds a 3-dimensional "constellation"
64 /// of data points.
65 ///
66 /// The constellation of points are used to compute magnetic hard/soft iron compensation terms.
67 /// The contents of this buffer are updated on a continuing basis.
68 struct MagBuffer
69 {
70  int16_t iBs[3][MAGBUFFSIZEX][MAGBUFFSIZEY]; ///< uncalibrated magnetometer readings
71  int32_t index[MAGBUFFSIZEX][MAGBUFFSIZEY]; ///< array of time indices
72  int16_t tanarray[MAGBUFFSIZEX - 1]; ///< array of tangents of (100 * angle)
73  int16_t iMagBufferCount; ///< number of magnetometer readings
74 };

76 /// Magnetic Calibration Structure
77 struct MagCalibration
78 {
79  // start of elements stored to flash memory on Save (16 * 4 = 64 bytes)
80  float fV[3]; ///< current hard iron offset x, y, z, (uT)
81  float finvW[3][3]; ///< current inverse soft iron matrix
82  float fB; ///< current geomagnetic field magnitude (uT)
83  float fBSq; ///< square of fB (uT^2)
84  float fFitErrorpc; ///< current fit error %
85  int32_t iValidMagCal; ///< solver used: 0 (no calibration) or 4, 7, 10 element
86  // end of elements stored to flash memory
87  // start of working elements not stored to flash memory
88  float ftrV[3]; ///< trial value of hard iron offset z, y, z (uT)
89  float ftrinvW[3][3]; ///< trial inverse soft iron matrix size
90  float ftrB; ///< trial value of geomagnetic field magnitude in uT
91  float ftrFitErrorpc; ///< trial value of fit error %
92  float fA[3][3]; ///< ellipsoid matrix A
93  float finvA[3][3]; ///< inverse of ellipsoid matrix A
94  float fmatA[10][10]; ///< scratch 10x10 float matrix used by calibration algorithms
95  float fmatB[10][10]; ///< scratch 10x10 float matrix used by calibration algorithms
96  float fvecA[10]; ///< scratch 10x1 vector used by calibration algorithms
97  float fvecB[4]; ///< scratch 4x1 vector used by calibration algorithms
98  float fYTY; ///< Y^T.Y for 4 element calibration = (iB^2)^2
99  int32_t iSumBs[3]; ///< sum of measurements in buffer (counts)
100  int32_t iMeanBs[3]; ///< average magnetic measurement (counts)
101  int32_t itimeslice; ///< counter for tine slicing magnetic calibration calculations
102  int8_t iCalInProgress; ///< flag denoting that a calibration is in progress
103  int8_t iNewCalibrationAvailable; ///< flag denoting that a new calibration has been computed
104  int8_t iInitiateMagCal; ///< flag to start a new magnetic calibration
105  int8_t iMagBufferReadOnly; ///< flag to denote that the magnetic measurement buffer is temporarily read only
106  int8_t i4ElementSolverTried; ///< flag to denote at least one attempt made with 4 element calibration
107  int8_t i7ElementSolverTried; ///< flag to denote at least one attempt made with 4 element calibration
108  int8_t i10ElementSolverTried; ///< flag to denote at least one attempt made with 4 element calibration
109 };
