83 /// This is a FreeRTOS (dual task) implementation of the NXP sensor fusion demo build.
84 int main(void)
85 {
86  ARM_DRIVER_I2C* I2Cdrv = &I2C_S_DRIVER; // defined in the <shield>.h file
87  BOARD_InitPins(); // defined in pin_mux.c, initializes pkg pins
88  BOARD_BootClockRUN(); // defined in clock_config.c, initializes clocks
89  BOARD_InitDebugConsole(); // defined in board.c, initializes the OpenSDA port
90 
91  I2Cdrv->Initialize(I2C_S_SIGNAL_EVENT); // Initialize the KSDK driver for the I2C port
92  I2Cdrv->PowerControl(ARM_POWER_FULL); // Set the I2C Power mode.
93  I2Cdrv->Control(ARM_I2C_BUS_SPEED, ARM_I2C_BUS_SPEED_FAST); // Configure the I2C bus speed
94 
95  initializeControlPort(&controlSubsystem); // configure pins and ports for the control sub-system
96  initializeStatusSubsystem(&statusSubsystem); // configure pins and ports for the status sub-system
97  initSensorFusionGlobals(&sfg, &statusSubsystem, &controlSubsystem); // Initialize sensor fusion structures
98  // "install" the sensors we will be using
99  sfg.installSensor(&sfg, &sensors[0], FXAS21002_I2C_ADDR, 1, (void*) I2Cdrv, &i2cBusInfo, FXAS21002_Init, FXAS21002_Read);
100  sfg.installSensor(&sfg, &sensors[1], MAG3110_I2C_ADDR, 1, (void*) I2Cdrv, &i2cBusInfo, MAG3110_Init, MAG3110_Read);
101  sfg.installSensor(&sfg, &sensors[2], MPL3115_I2C_ADDR, 1, (void*) I2Cdrv, &i2cBusInfo, MPL3115_Init, MPL3115_Read);
102  sfg.installSensor(&sfg, &sensors[3], FXLS8962_I2C_ADDR, 1, (void*) I2Cdrv, &i2cBusInfo, FXLS8962_Init, FXLS8962_Read);
103  sfg.initializeFusionEngine(&sfg); // This will initialize sensors and magnetic calibration
104 
105  event_group = xEventGroupCreate();
106  xTaskCreate(read_task, "READ", configMINIMAL_STACK_SIZE, NULL, tskIDLE_PRIORITY + 2, NULL);
107  xTaskCreate(fusion_task, "FUSION", configMINIMAL_STACK_SIZE, NULL, tskIDLE_PRIORITY + 1, NULL);
108 
109  sfg.setStatus(&sfg, NORMAL); // If we got this far, let's set status state to NORMAL
110  vTaskStartScheduler(); // Start the RTOS scheduler
111  sfg.setStatus(&sfg, HARD_FAULT); // If we got this far, FreeRTOS does not have enough memory allocated
112  for (;;) ;
113 }

115 static void read_task(void *pvParameters)
116 {
117  uint16_t i=0; // general counter variable
118  portTickType lastWakeTime;
119  const portTickType frequency = 1; // tick counter runs at the read rate
120  lastWakeTime = xTaskGetTickCount();
121  while (1)
122  {
123  for (i=1; i<=OVERSAMPLE_RATE; i++) {
124  vTaskDelayUntil(&lastWakeTime, frequency);
125  sfg.readSensors(&sfg, i); // Reads sensors, applies HAL and does averaging (if applicable)
126  }
127  xEventGroupSetBits(event_group, B0);
128  }
129 }
130 
131 static void fusion_task(void *pvParameters)
132 {
133  uint16_t i=0; // general counter variable
134  while (1)
135  {
136  xEventGroupWaitBits(event_group, /* The event group handle. */
137  B0, /* The bit pattern the event group is waiting for. */
138  pdTRUE, /* BIT_0 and BIT_4 will be cleared automatically. */
139  pdFALSE, /* Don't wait for both bits, either bit unblock task. */
140  portMAX_DELAY); /* Block indefinitely to wait for the condition to be met. */
141 
142  sfg.conditionSensorReadings(&sfg); // magCal is run as part of this
143  sfg.runFusion(&sfg); // Run the actual fusion algorithms
144  sfg.applyPerturbation(&sfg); // apply debug perturbation (testing only)
145 
146  sfg.loopcounter++; // The loop counter is used to "serialize" mag cal operations
147  i=i+1;
148  if (i>=4) { // Some status codes include a "blink" feature. This loop
149  i=0; // should cycle at least four times for that to operate correctly.
150  sfg.updateStatus(&sfg); // This is where pending status updates are made visible
151  }
152  sfg.queueStatus(&sfg, NORMAL); // assume NORMAL status for next pass through the loop
153  sfg.pControlSubsystem->stream(&sfg, sUARTOutputBuffer); // Send stream data to the Sensor Fusion Toolbox
154  }
155 }
