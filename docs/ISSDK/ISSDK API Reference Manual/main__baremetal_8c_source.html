73 /// This is a bare-metal implementation of the NXP sensor fusion demo build.
74 int main(void)
75 {
76  ARM_DRIVER_I2C* I2Cdrv = &I2C_S_DRIVER; // defined in the <shield>.h file
77  uint16_t i=0; // general counter variable
78  BOARD_InitPins(); // defined in pin_mux.c, initializes pkg pins
79  BOARD_BootClockRUN(); // defined in clock_config.c, initializes clocks
80  BOARD_InitDebugConsole(); // defined in board.c, initializes the OpenSDA port
81 
82  I2Cdrv->Initialize(I2C_S_SIGNAL_EVENT); // Initialize the KSDK driver for the I2C port
83  I2Cdrv->PowerControl(ARM_POWER_FULL); // Set the I2C Power mode.
84  I2Cdrv->Control(ARM_I2C_BUS_SPEED, ARM_I2C_BUS_SPEED_FAST); // Configure the I2C bus speed
85 
86  initializeControlPort(&controlSubsystem); // configure pins and ports for the control sub-system
87  initializeStatusSubsystem(&statusSubsystem); // configure pins and ports for the status sub-system
88  initSensorFusionGlobals(&sfg, &statusSubsystem, &controlSubsystem); // Initialize sensor fusion structures
89  // "install" the sensors we will be using
90 #if F_USING_ACCEL || F_USING_MAG
91  sfg.installSensor(&sfg, &sensors[0], FXOS8700_I2C_ADDR, 1, (void*) I2Cdrv, &i2cBusInfo, FXOS8700_Init, FXOS8700_Read);
92 #endif
93 #if F_USING_GYRO
94  sfg.installSensor(&sfg, &sensors[1], FXAS21002_I2C_ADDR, 1, (void*) I2Cdrv, &i2cBusInfo, FXAS21002_Init, FXAS21002_Read);
95 #endif
96 #if F_USING_PRESSURE
97  sfg.installSensor(&sfg, &sensors[2], MPL3115_I2C_ADDR, 1, (void*) I2Cdrv, &i2cBusInfo, MPL3115_Init, MPL3115_Read);
98 #endif
99  sfg.initializeFusionEngine(&sfg); // This will initialize sensors and magnetic calibration
100 
101  pit_init(1000000/FUSION_HZ); // pitIsrFlag will be set true at FUSION_HZ periodic intervals
102 
103  sfg.setStatus(&sfg, NORMAL); // If we got this far, let's set status state to NORMAL
104  while (true)
105  {
106  if (true == pitIsrFlag) { // Check whether occur interupt and toggle LED
107  sfg.readSensors(&sfg, 1); // Reads sensors, applies HAL and does averaging (if applicable)
108  sfg.conditionSensorReadings(&sfg); // magCal is run as part of this
109  sfg.runFusion(&sfg); // Run the actual fusion algorithms
110  sfg.applyPerturbation(&sfg); // apply debug perturbation (testing only)
111  sfg.loopcounter++; // The loop counter is used to "serialize" mag cal operations
112  i=i+1;
113  if (i>=4) { // Some status codes include a "blink" feature. This loop
114  i=0; // should cycle at least four times for that to operate correctly.
115  sfg.updateStatus(&sfg); // This is where pending status updates are made visible
116  }
117 
118  sfg.queueStatus(&sfg, NORMAL); // assume NORMAL status for next pass through the loop
119  sfg.pControlSubsystem->stream(&sfg, sUARTOutputBuffer); // Send stream data to the Sensor Fusion Toolbox
120  pitIsrFlag = false; // Reset the flag for the next cycle
121  }
122  }
123 }
