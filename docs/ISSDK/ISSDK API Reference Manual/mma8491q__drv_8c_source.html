44 int32_t MMA8491Q_I2C_Initialize(mma8491q_i2c_sensorhandle_t *pSensorHandle,
45  ARM_DRIVER_I2C *pBus,
46  uint8_t index,
47  uint16_t sAddress)
48 {
49  int32_t status;
50  MMA8491Q_STATUS_t reg = {.w = MMA8491Q_STATUS_RESERVED_MASK}; /* Set the reserved bits to 1. */
51 
52  /*! Check the input parameters. */
53  if ((pSensorHandle == NULL) || (pBus == NULL))
54  {
55  return SENSOR_ERROR_INVALID_PARAM;
56  }
57 
58  pSensorHandle->deviceInfo.deviceInstance = index;
59  pSensorHandle->deviceInfo.functionParam = NULL;
60  pSensorHandle->deviceInfo.idleFunction = NULL;
61 
62  /* Since this sensor does not have who am i register, we verify Status Register Reserved bits to validate
63  * communication. */
64  status = Register_I2C_Read(pBus, &pSensorHandle->deviceInfo, sAddress, MMA8491Q_STATUS, 1, (uint8_t *)&reg);
65  if ((ARM_DRIVER_OK != status) || (MMA8491Q_STATUS_RESERVED_ZERO != reg.b.reserved))
66  {
67  pSensorHandle->isInitialized = false;
68  return SENSOR_ERROR_INIT;
69  }
70 
71  /*! Initialize the sensor handle. */
72  pSensorHandle->pCommDrv = pBus;
73  pSensorHandle->slaveAddress = sAddress;
74  pSensorHandle->isInitialized = true;
75  return SENSOR_ERROR_NONE;
76 }

86 int32_t MMA8491Q_I2C_ReadData(mma8491q_i2c_sensorhandle_t *pSensorHandle,
87  const registerreadlist_t *pReadList,
88  uint8_t *pBuffer)
89 {
90  int32_t status;
91 
92  /*! Validate for the correct handle and register read list.*/
93  if ((pSensorHandle == NULL) || (pReadList == NULL) || (pBuffer == NULL))
94  {
95  return SENSOR_ERROR_INVALID_PARAM;
96  }
97 
98  /*! Check whether sensor handle is initialized before reading sensor data.*/
99  if (pSensorHandle->isInitialized != true)
100  {
101  return SENSOR_ERROR_INIT;
102  }
103 
104  /*! Parse through the read list and read the data one by one. */
105  status = Sensor_I2C_Read(pSensorHandle->pCommDrv, &pSensorHandle->deviceInfo, pSensorHandle->slaveAddress,
106  pReadList, pBuffer);
107  if (ARM_DRIVER_OK != status)
108  {
109  return SENSOR_ERROR_READ;
110  }
111 
112  return SENSOR_ERROR_NONE;
113 }
