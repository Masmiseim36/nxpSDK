98 void mma8491q_en_callback(void)

102  gSensorContext.inTransition = false;

105 /* Function for MMA8491Q specific initialization tasks. */
106 void mma8491q_mode_fwk_init(mma8491q_context_t *pSensorContext, void *pEN)
107 {
108  if (pSensorContext == NULL || pEN == NULL)
109  {
110  PRINTF("\r\n Error initializing MMA8491Q state framework. \r\n");
111  return;
112  }
113 
114  lptmr_config_t lptmrConfig;
115  /* Initialize EN Timer. */
116  LPTMR_GetDefaultConfig(&lptmrConfig);
117  LPTMR_Init(LPTMR0, &lptmrConfig);
118  LPTMR_EnableInterrupts(LPTMR0, kLPTMR_TimerInterruptEnable);
119  EnableIRQ(LPTMR0_IRQn);
120 
121  /* In case required time between 2 samples - T_ON is greater than the RESET Time of sensor,
122  * the ODR rate will be equal to the requested ODR - T_ON.
123  * In case required time between 2 samples - T_ON is less than the RESET Time of sensor,
124  * the ODR rate has to be at lest equal to the RESET Time for reliable operation. */
125  pSensorContext->adjustedODR =
126  (MMA8491Q_T_ODR_ms - MMA8491Q_T_ON_TYPICAL - LPTMR_T_START_STOP > MMA8491Q_T_RST_MIN) ?
127  MMA8491Q_T_ODR_ms - MMA8491Q_T_ON_TYPICAL - LPTMR_T_START_STOP :
128  MMA8491Q_T_RST_MIN;
129 
130  /* Initialize the Timer timeout variable. */
131  pSensorContext->inTransition = false;
132  /* Capture the GPIO handle. */
133  pSensorContext->pGpioEN = pEN;
134 
135  /* Put the Sensor into Active Mode to enable I2C communication. */
136  pSensorContext->inTransition = true;
137  LPTMR_SetTimerPeriod(LPTMR0, MSEC_TO_COUNT(MMA8491Q_T_ON_TYPICAL, CLOCK_GetFreq(kCLOCK_LpoClk)));
138  LPTMR_StartTimer(LPTMR0);
139 
140  /* Set EN = 1 to put the Sensor into Active Mode. */
141  pGpioDriver->set_pin(pSensorContext->pGpioEN);
142  /* Wait until Sensor is in ACTIVE Mode. */
143  while (pSensorContext->inTransition)
144  { /* Put CPU into wait mode, the timer expiry interrupt will awake the CPU. */
145  SMC_SetPowerModeWait(SMC);
146  }
147 
148  pSensorContext->sensorMode = SENSOR_MODE_ACTIVE;
149 }

151 /* Function to send sensor into Active Mode. */
152 void mma8491q_mode_activate(mma8491q_context_t *pSensorContext)
153 {
154  if (pSensorContext == NULL || pSensorContext->sensorMode == SENSOR_MODE_STANDBY)
155  {
156  PRINTF("\r\n Error setting MMA8491Q to Active Mode. \r\n");
157  return;
158  }
159  /* Wait until Sensor has entered SHUTDOWN Mode. */
160  while (pSensorContext->inTransition)
161  { /* Put CPU into wait mode, the timer expiry interrupt will awake the CPU. */
162  SMC_SetPowerModeWait(SMC);
163  }
164  /* Start Timer for ACTIVE Mode. */
165  pSensorContext->inTransition = true;
166  LPTMR_SetTimerPeriod(LPTMR0, MSEC_TO_COUNT(MMA8491Q_T_ON_TYPICAL, CLOCK_GetFreq(kCLOCK_LpoClk)));
167  LPTMR_StartTimer(LPTMR0);
168 
169  /* Set EN = 1 to put the Sensor into Active Mode. */
170  pGpioDriver->set_pin(pSensorContext->pGpioEN);
171  /* Wait until Sensor is in ACTIVE Mode. */
172  while (pSensorContext->inTransition)
173  { /* Put CPU into wait mode, the timer expiry interrupt will awake the CPU. */
174  SMC_SetPowerModeWait(SMC);
175  }
176 
177  pSensorContext->sensorMode = SENSOR_MODE_STANDBY;
178 }
179 
180 /* Function to send sensor into Shutdown Mode. */
181 void mma8491q_mode_shutdown(mma8491q_context_t *pSensorContext)
182 {
183  if (pSensorContext == NULL)
184  {
185  PRINTF("\r\n Error setting MMA8491Q to Shutdown Mode. \r\n");
186  return;
187  }
188 
189  /* Start Timer for SHUTDOWN Mode. */
190  pSensorContext->inTransition = true;
191  LPTMR_SetTimerPeriod(LPTMR0, MSEC_TO_COUNT(pSensorContext->adjustedODR, CLOCK_GetFreq(kCLOCK_LpoClk)));
192  LPTMR_StartTimer(LPTMR0);
193 
194  /* Set EN = 0 to put the Sensor into Shutdown Mode. */
195  pGpioDriver->clr_pin(pSensorContext->pGpioEN);
196  pSensorContext->sensorMode = SENSOR_MODE_SHUTDOWN;
197 }

199 /*!
200  * @brief Main function
201  */
202 int main(void)
203 {
204  int32_t status;
205  uint8_t dataReady;
206  uint8_t data[MMA8491Q_DATA_SIZE];
207  mma8491q_acceldata_t rawData = {.timestamp = 0};
208 
209  ARM_DRIVER_I2C *I2Cdrv = &I2C_S_DRIVER; // Now using the shield.h value!!!
210  mma8491q_i2c_sensorhandle_t mma8491qDriver;
211 
212  BOARD_InitPins();
213  BOARD_BootClockRUN();
214  BOARD_SystickEnable();
215  BOARD_InitDebugConsole();
216 
217  PRINTF("\r\n ISSDK MMA8491Q sensor driver example demonstration with poll mode\r\n");
218 
219  /* Tilt Pin IDs. */
220  pGpioDriver->pin_init(&MMA8491_EN, GPIO_DIRECTION_OUT, NULL, NULL, NULL);
221  pGpioDriver->pin_init(&MMA8491_XOUT, GPIO_DIRECTION_IN, &gGpioConfigInPins, NULL, NULL);
222  pGpioDriver->pin_init(&MMA8491_YOUT, GPIO_DIRECTION_IN, &gGpioConfigInPins, NULL, NULL);
223  pGpioDriver->pin_init(&MMA8491_ZOUT, GPIO_DIRECTION_IN, &gGpioConfigInPins, NULL, NULL);
224 
225  /*! Initialize the I2C driver. */
226  status = I2Cdrv->Initialize(I2C_S_SIGNAL_EVENT);
227  if (ARM_DRIVER_OK != status)
228  {
229  PRINTF("\r\n I2C Initialization Failed\r\n");
230  return -1;
231  }
232 
233  /*! Set the I2C Power mode. */
234  status = I2Cdrv->PowerControl(ARM_POWER_FULL);
235  if (ARM_DRIVER_OK != status)
236  {
237  PRINTF("\r\n I2C Power Mode setting Failed\r\n");
238  return -1;
239  }
240 
241  /*! Set the I2C bus speed. */
242  status = I2Cdrv->Control(ARM_I2C_BUS_SPEED, ARM_I2C_BUS_SPEED_FAST);
243  if (ARM_DRIVER_OK != status)
244  {
245  PRINTF("\r\n I2C Control Mode setting Failed\r\n");
246  return -1;
247  }
248 
249  /*! Initialize MMA8491_EN Pin handling framework. */
250  mma8491q_mode_fwk_init(&gSensorContext, &MMA8491_EN);
251 
252  /* Initialize the Sensor Driver. */
253  status = MMA8491Q_I2C_Initialize(&mma8491qDriver, &I2C_S_DRIVER, I2C_S_DEVICE_INDEX, MMA8491_I2C_ADDR);
254  if (SENSOR_ERROR_NONE != status)
255  {
256  PRINTF("\r\n Sensor Initialization Failed\r\n");
257  return -1;
258  }
259  PRINTF("\r\n Successfully Initiliazed Sensor\r\n");
260 
261  /*! Set the task to be executed while waiting for I2C transactions to complete. */
262  MMA8491Q_I2C_SetIdleTask(&mma8491qDriver, (registeridlefunction_t)SMC_SetPowerModeWait, SMC);
263 
264  for (;;) /* Forever loop */
265  {
266  /* Put the Sensor in Active Mode. */
267  mma8491q_mode_activate(&gSensorContext);
268 
269  /*! Wait for data ready from the MMA8491Q. */
270  status = MMA8491Q_I2C_ReadData(&mma8491qDriver, cMma8491qStatus, &dataReady);
271  if (0 == (dataReady & MMA8491Q_STATUS_ZYXDR_MASK))
272  { /* Loop, if new sample is not available. */
273  continue;
274  }
275 
276  /*! Read the raw sensor data from the MMA8491Q. */
277  status = MMA8491Q_I2C_ReadData(&mma8491qDriver, cMma8491qOutput, data);
278  if (ARM_DRIVER_OK != status)
279  {
280  PRINTF("\r\n Read Failed. \r\n");
281  return -1;
282  }
283 
284  /* Put the sensor in Shutdown Mode. */
285  mma8491q_mode_shutdown(&gSensorContext);
286 
287  /* Read Tilt Output */
288  rawData.tilt[0] = pGpioDriver->read_pin(&MMA8491_XOUT);
289  rawData.tilt[1] = pGpioDriver->read_pin(&MMA8491_YOUT);
290  rawData.tilt[2] = pGpioDriver->read_pin(&MMA8491_ZOUT);
291 
292  /*! Process the sample and convert the raw sensor data. */
293  rawData.accel[0] = ((int16_t)data[0] << 8) | (data[1]);
294  rawData.accel[1] = ((int16_t)data[2] << 8) | (data[3]);
295  rawData.accel[2] = ((int16_t)data[4] << 8) | (data[5]);
296 
297  PRINTF("\r\n Tilt : X = %d Y = %d Z = %d ", rawData.tilt[0], rawData.tilt[1], rawData.tilt[2]);
298  PRINTF("|| Accel : X = %5d Y = %5d Z = %5d \r\n", rawData.accel[0], rawData.accel[1], rawData.accel[2]);
299  ASK_USER_TO_RESUME(1000); /* Ask for user input after processing 1000 samples. */
300  }
301 }
