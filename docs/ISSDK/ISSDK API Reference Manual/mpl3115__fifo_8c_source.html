56 //-----------------------------------------------------------------------
57 // Macros
58 //-----------------------------------------------------------------------
59 #define FIFO_WMRK_SIZE (8) /* Buffer 8 Samples. */
60 #define MPL3115_DATA_SIZE (5) /* 3 byte Pressure/Altitude and 2 byte Temperature. */
61 /*! In MPL3115 the Auto Acquisition Time Step (ODR) can be set only in powers of 2 (i.e. 2^x, where x is the
62  * SAMPLING_EXPONENT).
63  * This gives a range of 1 second to 2^15 seconds (9 hours). */
64 #define MPL3115_SAMPLING_EXPONENT (0) /* 1 second. */
65 
66 //-----------------------------------------------------------------------
67 // Constants
68 //-----------------------------------------------------------------------
69 /*! @brief Register settings for FIFO (buffered) mode @ default 1 sample per second. */
70 const registerwritelist_t cMpl3115ConfigFIFO[] = {
71  /* Set FIFO Mode and set FIFO Watermark Level. */
72  {MPL3115_F_SETUP, MPL3115_F_SETUP_F_MODE_STOP_MODE | FIFO_WMRK_SIZE,
73  MPL3115_F_SETUP_F_MODE_MASK | MPL3115_F_SETUP_F_WMRK_MASK},
74  /* Enable Data Ready and Event flags for Pressure, Temperature or either. */
75  {MPL3115_PT_DATA_CFG,
76  MPL3115_PT_DATA_CFG_TDEFE_ENABLED | MPL3115_PT_DATA_CFG_PDEFE_ENABLED | MPL3115_PT_DATA_CFG_DREM_ENABLED,
77  MPL3115_PT_DATA_CFG_TDEFE_MASK | MPL3115_PT_DATA_CFG_PDEFE_MASK | MPL3115_PT_DATA_CFG_DREM_MASK},
78  /* Set Over Sampling Ratio to 128. */
79  {MPL3115_CTRL_REG1, MPL3115_CTRL_REG1_OS_OSR_128, MPL3115_CTRL_REG1_OS_MASK},
80  {MPL3115_CTRL_REG2, MPL3115_SAMPLING_EXPONENT, MPL3115_CTRL_REG2_ST_MASK},
81  __END_WRITE_DATA__};
82 
83 /*! @brief Address of Status Register. */
84 const registerreadlist_t cMpl3115Status[] = {{.readFrom = MPL3115_STATUS, .numBytes = 1}, __END_READ_DATA__};
85 
86 /*! @brief Address and size of Pressure+Temperature Data in FIFO Mode. */
87 const registerreadlist_t cMpl3115OutputFIFO[] = {
88  {.readFrom = MPL3115_OUT_P_MSB, .numBytes = MPL3115_DATA_SIZE * FIFO_WMRK_SIZE}, __END_READ_DATA__};

90 //-----------------------------------------------------------------------
91 // Functions
92 //-----------------------------------------------------------------------
93 /*!
94  * @brief Main function
95  */
96 int main(void)
97 {
98  int16_t tempInDegrees;
99  uint32_t pressureInPascals;
100  int32_t status;
101  uint8_t dataReady;
102  uint8_t data[MPL3115_DATA_SIZE * FIFO_WMRK_SIZE];
103  mpl3115_pressuredata_t rawData;
104 
105  ARM_DRIVER_I2C *I2Cdrv = &I2C_S_DRIVER; // Now using the shield.h value!!!
106  mpl3115_i2c_sensorhandle_t mpl3115Driver;
107 
108  BOARD_InitPins();
109  BOARD_BootClockRUN();
110  BOARD_SystickEnable();
111  BOARD_InitDebugConsole();
112 
113  PRINTF("\r\n ISSDK MPL3115 sensor driver example demonstration with fifo mode\r\n");
114 
115  /*! Initialize the I2C driver. */
116  status = I2Cdrv->Initialize(I2C_S_SIGNAL_EVENT);
117  if (ARM_DRIVER_OK != status)
118  {
119  PRINTF("\r\n I2C Initialization Failed\r\n");
120  return -1;
121  }
122 
123  /*! Set the I2C Power mode. */
124  status = I2Cdrv->PowerControl(ARM_POWER_FULL);
125  if (ARM_DRIVER_OK != status)
126  {
127  PRINTF("\r\n I2C Power Mode setting Failed\r\n");
128  return -1;
129  }
130 
131  /*! Set the I2C bus speed. */
132  status = I2Cdrv->Control(ARM_I2C_BUS_SPEED, ARM_I2C_BUS_SPEED_FAST);
133  if (ARM_DRIVER_OK != status)
134  {
135  PRINTF("\r\n I2C Control Mode setting Failed\r\n");
136  return -1;
137  }
138 
139  /*! Initialize MPL3115 sensor driver. */
140  status = MPL3115_I2C_Initialize(&mpl3115Driver, &I2C_S_DRIVER, I2C_S_DEVICE_INDEX, MPL3115_I2C_ADDR,
141  MPL3115_WHOAMI_VALUE);
142  if (SENSOR_ERROR_NONE != status)
143  {
144  PRINTF("\r\n Sensor Initialization Failed\r\n");
145  return -1;
146  }
147  PRINTF("\r\n Successfully Initiliazed Sensor\r\n");
148 
149  /*! Set the task to be executed while waiting for I2C transactions to complete. */
150  MPL3115_I2C_SetIdleTask(&mpl3115Driver, (registeridlefunction_t)SMC_SetPowerModeWait, SMC);
151 
152  /*! Configure the MPL3115 sensor. */
153  status = MPL3115_I2C_Configure(&mpl3115Driver, cMpl3115ConfigFIFO);
154  if (SENSOR_ERROR_NONE != status)
155  {
156  PRINTF("\r\nMPL3115 now active and entering data read loop...\r\n");
157  return -1;
158  }
159  PRINTF("\r\n Successfully Applied MPL3115 Sensor Configuration\r\n");
160 
161  for (;;) /* Forever loop */
162  {
163  /*! Wait for data ready from the MPL3115. */
164  status = MPL3115_I2C_ReadData(&mpl3115Driver, cMpl3115Status, &dataReady);
165  if (0 == (dataReady & MPL3115_F_STATUS_F_WMKF_FLAG_MASK))
166  { /* Loop, if new sample is not available. */
167  continue;
168  }
169 
170  /*! Read new raw sensor data from the MPL3115. */
171  status = MPL3115_I2C_ReadData(&mpl3115Driver, cMpl3115OutputFIFO, data);
172  if (ARM_DRIVER_OK != status)
173  {
174  PRINTF("\r\n Read Failed. \r\n");
175  return -1;
176  }
177 
178  /* Reset Counters */
179  pressureInPascals = 0;
180  tempInDegrees = 0;
181 
182  for (uint8_t i = 0; i < FIFO_WMRK_SIZE; i++)
183  { /*! Process all samples and convert the raw sensor data. */
184  rawData.pressure = (uint32_t)((data[i * MPL3115_DATA_SIZE + 0]) << 16) |
185  ((data[i * MPL3115_DATA_SIZE + 1]) << 8) | ((data[i * MPL3115_DATA_SIZE + 2]));
186  pressureInPascals += rawData.pressure / MPL3115_PRESSURE_CONV_FACTOR;
187 
188  rawData.temperature = (int16_t)((data[i * MPL3115_DATA_SIZE + 3]) << 8) | (data[i * MPL3115_DATA_SIZE + 4]);
189  tempInDegrees += rawData.temperature / MPL3115_TEMPERATURE_CONV_FACTOR;
190  }
191  PRINTF("\r\nAverage Pressure = %d Pa\r\n", pressureInPascals / FIFO_WMRK_SIZE);
192  PRINTF("\r\nAverage Temperature = %d degC\r\n", tempInDegrees / FIFO_WMRK_SIZE);
193  ASK_USER_TO_RESUME(16 / FIFO_WMRK_SIZE); /* Ask for user input after processing 16 samples. */
194  }
195 }
