67  {MPL3115_CTRL_REG1, MPL3115_CTRL_REG1_ALT_ALT, MPL3115_CTRL_REG1_ALT_MASK},
68  __END_WRITE_DATA__};

79 /*! @brief Address and size of Raw Altitude+Temperature Data. */

83 //-----------------------------------------------------------------------
84 // Functions
85 //-----------------------------------------------------------------------
86 /*!
87  * @brief Main function
88  */
89 int main(void)
90 {
91  int16_t tempInDegrees;
92  int32_t altitudeInMeters;
93  int32_t status;
94  uint8_t dataReady;
95  uint8_t data[MPL3115_DATA_SIZE];
96  mpl3115_altitudedata_t rawData;
97 
98  ARM_DRIVER_I2C *I2Cdrv = &I2C_S_DRIVER; // Now using the shield.h value!!!
99  mpl3115_i2c_sensorhandle_t mpl3115Driver;
100  registerDeviceInfo_t deviceInfo;
101 
102  BOARD_InitPins();
103  BOARD_BootClockRUN();
104  BOARD_SystickEnable();
105  BOARD_InitDebugConsole();
106 
107  PRINTF("\r\n ISSDK MPL3115 sensor driver example demonstration with oneshot mode\r\n");
108 
109  /*! Initialize the I2C driver. */
110  status = I2Cdrv->Initialize(I2C_S_SIGNAL_EVENT);
111  if (ARM_DRIVER_OK != status)
112  {
113  PRINTF("\r\n I2C Initialization Failed\r\n");
114  return -1;
115  }
116 
117  /*! Set the I2C Power mode. */
118  status = I2Cdrv->PowerControl(ARM_POWER_FULL);
119  if (ARM_DRIVER_OK != status)
120  {
121  PRINTF("\r\n I2C Power Mode setting Failed\r\n");
122  return -1;
123  }
124 
125  /*! Set the I2C bus speed. */
126  status = I2Cdrv->Control(ARM_I2C_BUS_SPEED, ARM_I2C_BUS_SPEED_FAST);
127  if (ARM_DRIVER_OK != status)
128  {
129  PRINTF("\r\n I2C Control Mode setting Failed\r\n");
130  return -1;
131  }
132 
133  /*! Initialize MPL3115 sensor driver. */
134  status = MPL3115_I2C_Initialize(&mpl3115Driver, &I2C_S_DRIVER, I2C_S_DEVICE_INDEX, MPL3115_I2C_ADDR,
135  MPL3115_WHOAMI_VALUE);
136  if (SENSOR_ERROR_NONE != status)
137  {
138  PRINTF("\r\n Sensor Initialization Failed\r\n");
139  return -1;
140  }
141  PRINTF("\r\n Successfully Initiliazed Sensor\r\n");
142 
143  /*! We do not need to call MPL3115_I2C_Configure() in this case as we are going to read samples on demand.
144  * Instead we directly write register settings for One-Shot Mode... */
145  deviceInfo.deviceInstance = I2C_S_DEVICE_INDEX;
146  deviceInfo.functionParam = SMC;
147  deviceInfo.idleFunction = (registeridlefunction_t)SMC_SetPowerModeWait;
148  status = Sensor_I2C_Write(mpl3115Driver.pCommDrv, &deviceInfo, mpl3115Driver.slaveAddress, cMpl3115ConfigAltitude);
149  if (ARM_DRIVER_OK != status)
150  {
151  PRINTF("\r\nWrite Failed.\r\n");
152  return -1;
153  }
154  PRINTF("\r\nMPL3115 will now trigger acquisition of one sample and then read the data...\r\n");
155 
156  /*! The code contained withing the braces illustrate steps to read one-sample.
157  * These can be repated at desired intervals to acquire more samples. */
158  { /* Trigger acquisition of One Sample. */
159  status = Sensor_I2C_Write(mpl3115Driver.pCommDrv, &deviceInfo, mpl3115Driver.slaveAddress, cMpl3115SetOST);
160  if (ARM_DRIVER_OK != status)
161  {
162  PRINTF("\r\nWrite Failed.\r\n");
163  return SENSOR_ERROR_WRITE;
164  }
165 
166  do /*! Keep checking the OST FLAG for completion. */
167  {
168  status = MPL3115_I2C_ReadData(&mpl3115Driver, cMpl3115GetOST, &dataReady);
169  if (ARM_DRIVER_OK != status)
170  {
171  PRINTF("\r\nRead Failed.\r\n");
172  return -1;
173  }
174  } /* Loop, untill sample acquisition is not completed. */
175  while (0 != (dataReady & MPL3115_CTRL_REG1_OST_MASK));
176 
177  /*! Read raw sensor data from the MPL3115. */
178  status = MPL3115_I2C_ReadData(&mpl3115Driver, cMpl3115OutputNormal, data);
179  if (ARM_DRIVER_OK != status)
180  {
181  PRINTF("\r\n Read Failed. \r\n");
182  return -1;
183  }
184 
185  /*! Process the sample and convert the raw sensor data. */
186  rawData.altitude = (int32_t)((data[0]) << 24) | ((data[1]) << 16) | ((data[2]) << 8);
187  rawData.temperature = (int16_t)((data[3]) << 8) | (data[4]);
188  altitudeInMeters = rawData.altitude / MPL3115_ALTITUDE_CONV_FACTOR;
189  tempInDegrees = rawData.temperature / MPL3115_TEMPERATURE_CONV_FACTOR;
190 
191  PRINTF("\r\nAltitude = %d Meters\r\n", altitudeInMeters);
192  PRINTF("\r\nTemperature = %d degC\r\n", tempInDegrees);
193  ASK_USER_TO_RESUME(1); /* Ask for user input after processing 8 samples. */
194  }
195 
196  return 0;
197 }
