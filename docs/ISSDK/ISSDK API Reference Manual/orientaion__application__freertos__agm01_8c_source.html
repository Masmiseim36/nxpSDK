127 /* Handler for Device Info and Streaming Control Commands. */
128 bool process_host_command(
129  uint8_t tag, uint8_t *hostCommand, uint8_t *hostResponse, size_t *hostMsgSize, size_t respBufferSize)
130 {
131  bool success = false;
132 
133  /* If it is Host requesting Device Info, send Board Name and Shield Name. */
134  if (tag == HOST_PRO_INT_DEV_TAG)
135  { /* Byte 1 : Payload - Length of APPLICATION_NAME (b)
136  * Bytes=b : Payload Application Name
137  * Byte b+1 : Payload - Length of BOARD_NAME (s)
138  * Bytes=s : Payload Board Name
139  * Byte b+s+2 : Payload - Length of SHIELD_NAME (v)
140  * Bytes=v : Payload Shield Name */
141 
142  size_t appNameLen = strlen(embAppName);
143  size_t boardNameLen = strlen(boardString);
144  size_t shieldNameLen = strlen(shieldString);
145 
146  if (respBufferSize >= boardNameLen + shieldNameLen + appNameLen + 3)
147  { /* We have sufficient buffer. */
148  *hostMsgSize = 0;
149  }
150  else
151  {
152  return false;
153  }
154 
155  hostResponse[*hostMsgSize] = appNameLen;
156  *hostMsgSize += 1;
157 
158  memcpy(hostResponse + *hostMsgSize, embAppName, appNameLen);
159  *hostMsgSize += appNameLen;
160 
161  hostResponse[*hostMsgSize] = boardNameLen;
162  *hostMsgSize += 1;
163 
164  memcpy(hostResponse + *hostMsgSize, boardString, boardNameLen);
165  *hostMsgSize += boardNameLen;
166 
167  hostResponse[*hostMsgSize] = shieldNameLen;
168  *hostMsgSize += 1;
169 
170  memcpy(hostResponse + *hostMsgSize, shieldString, shieldNameLen);
171  *hostMsgSize += shieldNameLen;
172 
173  return true;
174  }
175 
176  /* If it is Host sending Streaming Commands, take necessary actions. */
177  if ((tag == (HOST_PRO_INT_CMD_TAG | HOST_PRO_CMD_W_CFG_TAG)) &&
178  (*hostMsgSize == HOST_MSG_CMD_ACT_OFFSET - HOST_MSG_LEN_LSB_OFFSET))
179  { /* Byte 1 : Payload - Operation Code (Start/Stop Operation Code)
180  * Byte 2 : Payload - Stream ID (Target Stream for carrying out operation) */
181  switch (hostCommand[0]) /* Execute desired operation (Start/Stop) on the requested Stream. */
182  {
183  case HOST_CMD_START:
184  if (hostCommand[1] == gPrimaryStreamID && gStreamingEnabled == false)
185  {
186  gStreamingEnabled = true;
187  success = true;
188  }
189  break;
190  case HOST_CMD_STOP:
191  if (hostCommand[1] == gPrimaryStreamID && gStreamingEnabled == true)
192  {
193  gStreamingEnabled = false;
194  success = true;
195  }
196  break;
197  default:
198  break;
199  }
200  *hostMsgSize = 0; /* Zero payload in response. */
201  }
202 
203  return success;
204 }

206 /*!
207  * @brief freeRTOS read_task function
208  */
209 static void read_task(void *pvParameters)
210 {
211  uint16_t i = 0;
212  portTickType lastWakeTime;
213  const portTickType frequency = 1; /* tick counter runs at the read rate */
214  lastWakeTime = xTaskGetTickCount();
215  while (1)
216  {
217  for (i = 1; i <= OVERSAMPLE_RATE; i++)
218  {
219  vTaskDelayUntil(&lastWakeTime, frequency);
220  /*! Reads sensors, applies HAL and does averaging (if applicable) */
221  sfg.readSensors(&sfg, i);
222  }
223  xEventGroupSetBits(event_group, B0);
224  }
225 }

227 /*!
228  * @brief freeRTOS fusion_task function
229  */
230 static void fusion_task(void *pvParameters)
231 {
232  uint16_t i = 0;
233  while (1)
234  {
235  xEventGroupWaitBits(event_group, /* The event group handle. */
236  B0, /* The bit pattern the event group is waiting for. */
237  pdTRUE, /* BIT_0 and BIT_4 will be cleared automatically. */
238  pdFALSE, /* Don't wait for both bits, either bit unblock task. */
239  portMAX_DELAY); /* Block indefinitely to wait for the condition to be met. */
240 
241  sfg.conditionSensorReadings(&sfg); /* magCal is run as part of this */
242  sfg.runFusion(&sfg); /* Run the actual fusion algorithms */
243  sfg.applyPerturbation(&sfg); /* apply debug perturbation (testing only) */
244 
245  sfg.loopcounter++; /* The loop counter is used to "serialize" mag cal operations */
246  i = i + 1;
247  if (i >= 4)
248  { /* Some status codes include a "blink" feature. This loop */
249  i = 0; /* should cycle at least four times for that to operate correctly */
250  sfg.updateStatus(&sfg); /* This is where pending status updates are made visible */
251  }
252  sfg.queueStatus(&sfg, NORMAL); /* assume NORMAL status for next pass through the loop */
253 
254  /* Check for incoming commands form Host. */
255  Host_IO_Receive(process_host_command, HOST_FORMAT_HDLC);
256 
257  sfg.pControlSubsystem->stream(&sfg,
258  orientOutputBuffer); /*! Encode Orietantion Stream Packet and send over Host I/O*/
259  }
260 }

319 /*!
320  * @brief Function to create orient packets for serial streaming over Host I/O
321  */
322 void encodeOrientPacketStream(SensorFusionGlobals *sfg, uint8_t *sUARTOutputBuffer)
323 {
324  Quaternion fq; /* quaternion to be transmitted */
325  static uint32_t iTimeStamp = 0; /* 1MHz time stamp */
326  uint16_t iIndex; /* output buffer counter */
327  int16_t scratch16; /* scratch int16_t */
328  int16_t iPhi, iThe, iRho; /* integer angles to be transmitted */
329  int16_t iDelta; /* magnetic inclination angle if available */
330  int16_t iOmega[3]; /* scaled angular velocity vector */
331  uint16_t isystick; /* algorithm systick time */
332  uint8_t tmpuint8_t; /* scratch uint8_t */
333  uint8_t flags; /* byte of flags */
334  size_t sUARTOutputBufLen = STREAMING_PAYLOAD_LEN;
335 
336  /* Update the 1 microsecond timestamp counter */
337  iTimeStamp += 1000000 / FUSION_HZ;
338 
339  /* cache local copies of control flags so we don't have to keep dereferencing pointers below */
340  quaternion_type quaternionPacketType;
341  quaternionPacketType = sfg->pControlSubsystem->QuaternionPacketType;
342 
343  /* Start the index to store orientation packet payload into the transmit buffer */
344  iIndex = STREAMING_HEADER_LEN;
345 
346  /*! Add 1 microsecond timestamp counter (4 bytes) value to Orientation Buffer */
347  updateOrientBuf(sUARTOutputBuffer, &iIndex, (uint8_t *)&iTimeStamp, 4);
348 
349  /* initialize default quaternion, flags byte, angular velocity and orientation */
350  fq.q0 = 1.0F;
351  fq.q1 = fq.q2 = fq.q3 = 0.0F;
352  flags = 0x00;
353  iOmega[CHX] = iOmega[CHY] = iOmega[CHZ] = 0;
354  iPhi = iThe = iRho = iDelta = 0;
355  isystick = 0;
356 
357  /* set the quaternion, flags, angular velocity and Euler angles */
358  if (sfg->iFlags & F_9DOF_GBY_KALMAN)
359  {
360  flags |= 0x08;
361  iDelta = (int16_t)(10.0F * sfg->SV_9DOF_GBY_KALMAN.fDeltaPl);
362  readAlgoParams((SV_ptr)&sfg->SV_9DOF_GBY_KALMAN, &fq, &iPhi, &iThe, &iRho, iOmega, &isystick);
363  }
364 
365  /*! Scale the quaternion and add to the Orientation Buffer */
366  scratch16 = (int16_t)(fq.q0 * 30000.0F);
367  updateOrientBuf(sUARTOutputBuffer, &iIndex, (uint8_t *)&scratch16, 2);
368  scratch16 = (int16_t)(fq.q1 * 30000.0F);
369  updateOrientBuf(sUARTOutputBuffer, &iIndex, (uint8_t *)&scratch16, 2);
370  scratch16 = (int16_t)(fq.q2 * 30000.0F);
371  updateOrientBuf(sUARTOutputBuffer, &iIndex, (uint8_t *)&scratch16, 2);
372  scratch16 = (int16_t)(fq.q3 * 30000.0F);
373  updateOrientBuf(sUARTOutputBuffer, &iIndex, (uint8_t *)&scratch16, 2);
374 
375 /* Set the coordinate system bits in flags from default NED (00) */
376 #if THISCOORDSYSTEM == ANDROID
377  /* Set the Android flag bits */
378  flags |= 0x10;
379 #elif THISCOORDSYSTEM == WIN8
380  /* set the Win8 flag bits */
381  flags |= 0x20;
382 #endif // THISCOORDSYSTEM
383 
384  /*! Add the co-ordinate info byte to the Orientation Buffer */
385  updateOrientBuf(sUARTOutputBuffer, &iIndex, &flags, 1);
386 
387  /*! Add the shield (bits 7-5) and Kinetis (bits 4-0) byte to the Orientation Buffer */
388  tmpuint8_t = ((THIS_SHIELD & 0x07) << 5) | (THIS_BOARD & 0x1F);
389  updateOrientBuf(sUARTOutputBuffer, &iIndex, &tmpuint8_t, 1);
390 
391  /*! Add Software version number to the Orientation Buffer */
392  scratch16 = THISBUILD;
393  updateOrientBuf(sUARTOutputBuffer, &iIndex, (uint8_t *)&scratch16, 2);
394 
395  /*! Add systick count to the Orientation Buffer */
396  updateOrientBuf(sUARTOutputBuffer, &iIndex, (uint8_t *)&isystick, 2);
397 
398  /*! Send packet to host only if streaming has been enabled by Host. */
399  if (gStreamingEnabled)
400  {
401  /*! Add Host I/O ISO streaming header */
402  Host_IO_Add_ISO_Header(gPrimaryStreamID, sUARTOutputBuffer, sUARTOutputBufLen);
403 
404  /*! Send Orientation streaming packet to Host using Host I/O */
405  Host_IO_Send(sUARTOutputBuffer, STREAMING_HEADER_LEN + sUARTOutputBufLen, HOST_FORMAT_HDLC);
406  }
407 
408  return;
409 }

470  FXAS21002_Read);
471 
472  /*! Initialize streaming and assign Stream IDs. */

474  /* Registering the FXOS8700 slave address with Host I/O for Register Read/Write. */
475  gPrimaryStreamID =
476  Host_IO_Init(pUartDriver, (void *)sensors[0].bus_driver, &sensors[0].deviceInfo, NULL, FXOS8700_I2C_ADDR);
477  /* Confirm if a valid Stream ID has been allocated for this stream. */
478  if (0 == gPrimaryStreamID)
479  {
480  return -1;
481  }
482 
483  /* Registering the FXAS21002 slave address with Host I/O for Register Read/Write. */
484  secondaryStreamID =
485  Host_IO_Init(pUartDriver, (void *)sensors[1].bus_driver, &sensors[1].deviceInfo, NULL, FXAS21002_I2C_ADDR);
486  /* Confirm if a valid Stream ID has been allocated for this stream. */
487  if (0 == secondaryStreamID)
488  {
489  return -1;
490  }
491 
492  /*! Initialize sensor fusion engine */
493  sfg.initializeFusionEngine(&sfg);
494 
495  /*! Create RTOS tasks for sensor fusion */
496  event_group = xEventGroupCreate();
497  xTaskCreate(read_task, "READ", configMINIMAL_STACK_SIZE, NULL, tskIDLE_PRIORITY + 2, NULL);
498  xTaskCreate(fusion_task, "FUSION", configMINIMAL_STACK_SIZE, NULL, tskIDLE_PRIORITY + 1, NULL);
499 
500  /*! Set sensor fusion engine status to NORMAL */
501  sfg.setStatus(&sfg, NORMAL);
502 
503  /*! Start RTOS Scheduler */
504  vTaskStartScheduler();
505 
506  sfg.setStatus(&sfg, HARD_FAULT); /* If we got this far, FreeRTOS does not have enough memory allocated */
507  for (;;)
508  ;
509 }
