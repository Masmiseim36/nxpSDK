139 /*
140 ** ===================================================================
141 ** Method : pedometer_run
142 ** brief : The interface function excutes the pedometer algorithm.
143 ** params[in] : pedometer_t *pPedometer, handle to the pedometer.
144 ** params[in] : ped_accel_t data acceleration data.
145 ** ===================================================================
146 */
147 int32_t pedometer_run(pedometer_t *pPedometer, ped_accel_t *pData)
148 {
149  uint32_t events = 0;
150  /* if suspend, don't execute the alogrithm and save the power.*/
151  bool suspend = suspend_compute(pPedometer, pData);
152  if (!suspend)
153  {
154  events = KeynetikHandleIncomingEvent(pData->accel[0], pData->accel[1], pData->accel[2]);
155  }
156  status_update(pPedometer, events, suspend);
157  return events;
158 }
159 /*
160 ** ===================================================================
161 ** Method : status_update
162 ** brief : update the status of the pedometer output
163 ** params[in] : pedometer_t *pPedometer, handle to the pedometer.
164 ** params[in] : uint32_t events kynetic event
165 ** params[in] : bool suspend suspend flag
166 ** ===================================================================
167 */
168 static void status_update(pedometer_t *pPedometer, uint32_t events, bool suspend)
169 {
170  bool activity_stable =
171  debounce_count(pPedometer->config.bits.activity_dbcntm, !(events & KEYNETIK_ACTIVITYCHANGED),
172  &pPedometer->private.activitycount, pPedometer->config.activitycount_threshold);
173 
174  activitylevel_t activity = activity_stable ? keynetikActivityLevel : pPedometer->status.status.bits.activity;
175  uint16_t stepcount = uint32_to_uint16(keynetikStepCount);
176  uint8_t newstatus = (activity << PEDOMETER_STATUS_ACTIVITY_BITNUM) & PEDOMETER_STATUS_ACTIVITY_MASK;
177 
178  if ((newstatus ^ pPedometer->status.status.byte) & PEDOMETER_STATUS_ACTIVITY_MASK)
179  newstatus |= PEDOMETER_STATUS_ACTCHG_MASK;
180 
181  if (pPedometer->config.stepcoalesce &&
182  (stepcount - pPedometer->private.stepchg_stepcount >= pPedometer->config.stepcoalesce))
183  {
184  pPedometer->private.stepchg_stepcount = stepcount;
185  newstatus |= PEDOMETER_STATUS_STEPCHG_MASK;
186  }
187 
188  if (suspend)
189  newstatus |= PEDOMETER_STATUS_SUSPEND_MASK;
190 
191  if ((newstatus ^ pPedometer->status.status.byte) & PEDOMETER_STATUS_SUSPEND_MASK)
192  newstatus |= PEDOMETER_STATUS_SUSPCHG_MASK;
193 
194  if (newstatus & PEDOMETER_STATUS_CHG_MASK)
195  newstatus |= PEDOMETER_STATUS_MRGFLG_MASK;
196  /* Update the status such stepcount, distance, speed etc..*/
197  pPedometer->status.stepcount = uint32_to_uint16(keynetikStepCount);
198  pPedometer->status.distance = uint32_to_uint16(keynetikDistance);
199  pPedometer->status.speed = keynetikSpeed;
200  pPedometer->status.calories = uint32_to_uint16(keynetikCalories);
201  pPedometer->status.status.byte = newstatus;
202 }
203 /*
204 ** ===================================================================
205 ** Method : suspend_compute
206 ** brief : compute the suspend status.
207 ** params[in] : pedometer_t *pPedometer, handle to the pedometer.
208 ** params[in] : ped_accel_t data acceleration data.
209 ** ===================================================================
210 */
211 static bool suspend_compute(pedometer_t *pPedometer, ped_accel_t *pData)
212 {
213  /* compute the magnitude*/
214  uint32_t mag2 = SQUARED(pData->accel[0]) + SQUARED(pData->accel[1]) + SQUARED(pData->accel[2]);
215  bool stationary =
216  (mag2 > SQUARED(pPedometer->config.sleepminimum)) && (mag2 < SQUARED(pPedometer->config.sleepmaximum));
217  bool suspend = debounce_count(pPedometer->config.bits.sleep_dbcntm, stationary, &pPedometer->status.sleepcount,
218  pPedometer->config.sleepcount_threshold);
219  return (suspend);
220 }
221 /*
222 ** ===================================================================
223 ** Method : uint32_to_uint16
224 ** brief : conversion function from 32 to 16.
225 ** params[in] : uint32_t x, imput
226 ** params[in] : ped_accel_data data acceleration data.
227 ** ===================================================================
228 */
229 uint16_t uint32_to_uint16(uint32_t val)
230 {
231  if (val > 0x0000FFFF)
232  return (0xFFFF);
233  return ((uint16_t)val);
234 }
235 /*
236 ** ===================================================================
237 ** Method : debounce_count
238 ** brief : compute the debouct status.
239 ** params[in] : bool dbcntm,
240 ** params[in] : bool condition
241 ** params[in] : debounce_count_t *count,
242 ** params[in] : debounce_count_t threshold, debouce threshold
243 ** ===================================================================
244 */
245 static bool debounce_count(bool dbcntm, bool condition, debounce_count_t *count, debounce_count_t threshold)
246 {
247  if (condition)
248  {
249  /* increment count up to, but not above the threshold */
250  *count = (*count + 1) < threshold ? *count + 1 : threshold;
251  }
252  else
253  {
254  if (dbcntm)
255  *count = 0;
256  else
257  /* decrement count down to, but not below zero */
258  *count = (*count - 1) > 0 ? *count - 1 : 0;
259  }
260  return ((bool)(*count >= threshold));
261 }
