52 /// precision accelerometer calibration structure
53 typedef struct AccelCalibration
54 {
55  // start of elements stored in flash memory
56  float fV[3]; ///< offset vector (g)
57  float finvW[3][3]; ///< inverse gain matrix
58  float fR0[3][3]; ///< forward rotation matrix for measurement 0
59  // end of elements stored in flash memory
60  float fmatA[10][10]; ///< scratch 10x10 matrix used by calibration algorithms
61  float fmatB[10][10]; ///< scratch 10x10 matrix used by calibration algorithms
62  float fvecA[10]; ///< scratch 10x1 vector used by calibration algorithms
63  float fvecB[4]; ///< scratch 4x1 vector used by calibration algorithms
64  float fA[3][3]; ///< ellipsoid matrix A
65  float finvA[3][3]; ///< inverse of the ellipsoid matrix A
66 } AccelCalibration;
