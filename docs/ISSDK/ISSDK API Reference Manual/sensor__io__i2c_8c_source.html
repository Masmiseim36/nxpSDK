45 /*! The interface function to block write commands to a sensor. */
46 int32_t Sensor_I2C_BlockWrite(ARM_DRIVER_I2C *pCommDrv,
47  registerDeviceInfo_t *devInfo,
48  uint16_t slaveAddress,
49  const registercommandlist_t *pCommandList,
50  uint8_t error_mask)
51 {
52  int32_t status;
53  bool commandComplete;
54  uint8_t cocoBuffer[2] = {0};
55 
56  /*! Validate for the correct handle.*/
57  if ((pCommDrv == NULL) || (pCommandList == NULL))
58  {
59  return SENSOR_ERROR_BAD_ADDRESS;
60  }
61 
62  const registercommandlist_t *pCmd = pCommandList;
63  /*! Update register values based on register write list unless the next Cmd is the list terminator */
64  do
65  { /*! Write the command based on the values in the command and value pair.*/
66  status =
67  Register_I2C_BlockWrite(pCommDrv, devInfo, slaveAddress, pCmd->writeTo, pCmd->pWriteBuffer, pCmd->numBytes);
68  if (ARM_DRIVER_OK != status)
69  {
70  return SENSOR_ERROR_WRITE;
71  }
72 
73  do
74  { /*! Wait for Command Completion. */
75  BOARD_DELAY_ms(1);
76  status = Register_I2C_Read(pCommDrv, devInfo, slaveAddress, 0, sizeof(cocoBuffer), cocoBuffer);
77  if (ARM_DRIVER_OK != status)
78  {
79  return SENSOR_ERROR_READ;
80  }
81  commandComplete = cocoBuffer[1] & 0x80;
82  if (commandComplete && cocoBuffer[1] & error_mask)
83  {
84  return SENSOR_ERROR_WRITE;
85  }
86  } while (!commandComplete);
87  } while ((++pCmd)->writeTo != 0xFFFF);
88 
89  return SENSOR_ERROR_NONE;
90 }

92 /*! The interface function to write register data to a sensor. */
93 int32_t Sensor_I2C_Write(ARM_DRIVER_I2C *pCommDrv,
94  registerDeviceInfo_t *devInfo,
95  uint16_t slaveAddress,
96  const registerwritelist_t *pRegWriteList)
97 {
98  int32_t status;
99  bool repeatedStart;
100 
101  /*! Validate for the correct handle.*/
102  if ((pCommDrv == NULL) || (pRegWriteList == NULL))
103  {
104  return SENSOR_ERROR_BAD_ADDRESS;
105  }
106 
107  const registerwritelist_t *pCmd = pRegWriteList;
108 
109  /*! Update register values based on register write list unless the next Cmd is the list terminator */
110  do
111  {
112  repeatedStart = (pCmd + 1)->writeTo != 0xFFFF;
113  /*! Set the register based on the values in the register value pair.*/
114  status =
115  Register_I2C_Write(pCommDrv, devInfo, slaveAddress, pCmd->writeTo, pCmd->value, pCmd->mask, repeatedStart);
116  if (ARM_DRIVER_OK != status)
117  {
118  return SENSOR_ERROR_WRITE;
119  }
120  ++pCmd;
121  } while (repeatedStart);
122 
123  return SENSOR_ERROR_NONE;
124 }

126 /*! The interface function to read register data from a sensor. */
127 int32_t Sensor_I2C_Read(ARM_DRIVER_I2C *pCommDrv,
128  registerDeviceInfo_t *devInfo,
129  uint16_t slaveAddress,
130  const registerreadlist_t *pReadList,
131  uint8_t *pOutBuffer)
132 {
133  int32_t status;
134  uint8_t *pBuf;
135 
136  /*! Validate for the correct handle.*/
137  if (pCommDrv == NULL || pReadList == NULL || pOutBuffer == NULL)
138  {
139  return SENSOR_ERROR_BAD_ADDRESS;
140  }
141  const registerreadlist_t *pCmd = pReadList;
142 
143  /*! Traverse the read list and read the registers one by one unless the register read list numBytes is zero*/
144  for (pBuf = pOutBuffer; pCmd->numBytes != 0; pCmd++)
145  {
146  status = Register_I2C_Read(pCommDrv, devInfo, slaveAddress, pCmd->readFrom, pCmd->numBytes, pBuf);
147  if (ARM_DRIVER_OK != status)
148  {
149  return SENSOR_ERROR_READ;
150  }
151  pBuf += pCmd->numBytes;
152  }
153  return SENSOR_ERROR_NONE;
154 }
