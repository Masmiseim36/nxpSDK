42 
43 #define THISBUILD 700 ///< define build number sent in debug packet for display purposes only
44 
45 /// @name CoordinateSystemBitFields
46 /// These defines determine the frame of reference (x, y, z axes and Euler angles) standard
47 /// to be used for a particular build. Change THISCOORDSYSTEM to whichever of NED, ANDROID
48 /// or WIN8 you prefer.
49 ///@{
50 #define NED 0 ///< identifier for NED (Aerospace) axes and angles
51 #define ANDROID 1 ///< identifier for Android axes and angles
52 #define WIN8 2 ///< identifier for Windows 8 axes and angles
53 #define THISCOORDSYSTEM NED ///< the coordinate system to be used
54 ///@}
55 
56 ///@{
57 /// @name SensorBitFields
58 /// These bit-field values are used to declare which sensor types are used in the application.
59 /// Change bit-field values to 0x0000 for any features NOT USED
60 #define F_USING_ACCEL 0x0001 ///< nominally 0x0001 if an accelerometer is to be used, 0x0000 otherwise
61 #define F_USING_MAG 0x0002 ///< nominally 0x0002 if an magnetometer is to be used, 0x0000 otherwise
62 #define F_USING_GYRO 0x0004 ///< nominally 0x0004 if a gyro is to be used, 0x0000 otherwise
63 #define F_USING_PRESSURE 0x0008 ///< nominally 0x0008 if altimeter is to be used, 0x0000 otherwise
64 #define F_USING_TEMPERATURE 0x0010 ///< nominally 0x0010 if temp sensor is to be used, 0x0000 otherwise
65 #define F_ALL_SENSORS 0x001F ///< refers to all applicable sensor types for the given physical unit
66 ///@}
67 /// @name FusionSelectionBitFields
68 /// These bit-field values are used to declare which sensor fusion algorithms are used
69 /// in the application. You can use more than one, altough they all run from the same data.
70 /// Change individual bit-field values to 0x0000 for any features NOT USED.
71 ///@{
72 #define F_1DOF_P_BASIC 0x0100 ///< 1DOF pressure (altitude) and temperature algorithm selector - 0x0100 to include, 0x0000 otherwise
73 #define F_3DOF_G_BASIC 0x0200 ///< 3DOF accel tilt (accel) algorithm selector - 0x0200 to include, 0x0000 otherwise
74 #define F_3DOF_B_BASIC 0x0400 ///< 3DOF mag eCompass (vehicle/mag) algorithm selector - 0x0400 to include, 0x0000 otherwise
75 #define F_3DOF_Y_BASIC 0x0800 ///< 3DOF gyro integration algorithm selector - 0x0800 to include, 0x0000 otherwise
76 #define F_6DOF_GB_BASIC 0x1000 ///< 6DOF accel and mag eCompass algorithm selector - 0x1000 to include, 0x0000 otherwise
77 #define F_6DOF_GY_KALMAN 0x2000 ///< 6DOF accel and gyro (Kalman) algorithm selector - 0x2000 to include, 0x0000 otherwise
78 #define F_9DOF_GBY_KALMAN 0x4000 ///< 9DOF accel, mag and gyro algorithm selector - 0x4000 to include, 0x0000 otherwise
79 ///@}
80 
81 /// @name SensorParameters
82 /// FIFO sizes effect the size of the sensor data structures. ODR refers to "Output Data Rate"
83 ///@{
84 #define ACCEL_FIFO_SIZE 32 ///< FXOS8700 (accel), MMA8652, FXLS8952 all have 32 element FIFO
85 #define MAG_FIFO_SIZE 16 ///< FXOS8700 (mag), MAG3110 have no FIFO so equivalent to 1 element FIFO
86 #define GYRO_FIFO_SIZE 32 ///< FXAX21000, FXAS21002 have 32 element FIFO
87 #define ACCEL_ODR_HZ 200 ///< (int) requested accelerometer ODR Hz (over-rides MAG_ODR_HZ for FXOS8700)
88 #define MAG_ODR_HZ 200 ///< (int) requested magnetometer ODR Hz (over-ridden by ACCEL_ODR_HZ for FXOS8700)
89 #define GYRO_ODR_HZ 400 ///< (int) requested gyroscope ODR Hz
90 #define FUSION_HZ 40 ///< (int) actual rate of fusion algorithm execution and sensor FIFO reads
91 #define FAST_LOOP_HZ 80 ///< Over Sample Ratio * FUSION_HZ when using no FIFO
92 #define OVERSAMPLE_RATE FAST_LOOP_HZ/FUSION_HZ
93 ///@}
