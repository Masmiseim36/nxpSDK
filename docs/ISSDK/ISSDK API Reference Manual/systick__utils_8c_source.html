32  * @file systick_utils.c
33  * @brief Encapsulates the ARM sysTick counter, which is used for computing delays.
34  * ARM-core specific function that enables the ARM systick timer on Kinetis uCs.
35  * the timer is 24 bit so allows measurement of intervals up to 2^24/CORE_SYSTICK_HZ secs=0.35s for a 48MHz uC.

65 // ARM-core specific function to enable systicks.
66 void BOARD_SystickEnable(void)
67 {
68  SYST_CSR = SysTick_CTRL_CLKSOURCE_Msk | SysTick_CTRL_TICKINT_Msk |
69  SysTick_CTRL_ENABLE_Msk; // Enable systick from internal clock with Interrupts.
70  SYST_RVR = 0x00FFFFFFu; // Set reload to maximum 24 bit value.
71  return;
72 }

74 // ARM-core specific function to store the current systick timer ticks.
75 void BOARD_SystickStart(int32_t *pStart)
76 {
77  // Store the 24 bit systick timer.
78  g_ovf_stamp = SysTick_GetOVFCount();
79  *pStart = SYST_CVR & 0x00FFFFFF;
80 }
81 
82 // ARM-core specific function to compute the elapsed systick timer ticks.
83 int32_t BOARD_SystickElapsedTicks(int32_t *pStart)
84 {
85  int32_t elapsed;
86 
87  // Subtract the stored start ticks and check for wraparound down through zero.
88  elapsed = *pStart - (SYST_CVR & 0x00FFFFFF);
89  elapsed += SYST_RVR * (SysTick_GetOVFCount() - g_ovf_stamp);
90 
91  return elapsed;
92 }

94 // ARM-core specific function to compute the elapsed time in micro seconds.
95 uint32_t BOARD_SystickElapsedTime_us(int32_t *pStart)
96 {
97  uint32_t time_us, elapsed;
98  uint32_t systemCoreClock;
99 
100  elapsed = BOARD_SystickElapsedTicks(pStart);
101  systemCoreClock = CLOCK_GetFreq(kCLOCK_CoreSysClk);
102 
103  time_us = COUNT_TO_USEC(elapsed, systemCoreClock);
104 
105  // Update the 24 bit systick timer.
106  BOARD_SystickStart(pStart);
107 
108  return time_us;
109 }

111 // ARM-core specific function to insert delays in milli seconds.
112 void BOARD_DELAY_ms(uint32_t delay_ms)
113 {
114  int32_t start, elapsed;
115  uint32_t systemCoreClock;
116 
117  BOARD_SystickStart(&start);
118  do // Loop for requested number of ms.
119  {
120  elapsed = BOARD_SystickElapsedTicks(&start);
121  systemCoreClock = CLOCK_GetFreq(kCLOCK_CoreSysClk);
122  } while(COUNT_TO_MSEC(elapsed, systemCoreClock) < delay_ms);
123 }
