60 /*! @brief Function to compute the Elapsed systicks.
61  * @details This function computes the elapsed ticks by getting the difference
62  * of the current tick to the one in the arguement.
63  * @param[in] pStart Pointer to the variable contating the start systick.
64  * @return int32_t The elapsed systicks.
65  * @constraints None.
66  * @reeentrant Yes
67  */
68 int32_t BOARD_SystickElapsedTicks(int32_t *pStart);

70 /*! @brief Function to compute the Elapsed Time.

74  * @param[in] pStart Pointer to the variable contating the start systick.
75  * @return uint32_t The elapsed Time in micro seconds.
76  * @constraints None.
77  * @reeentrant Yes

79 uint32_t BOARD_SystickElapsedTime_us(int32_t *pStart);
